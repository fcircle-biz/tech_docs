仕様拡張
2クラス分の点数配列（classA、classB）を用意し、それぞれの平均点を計算し、どちらのクラスの平均点が高いかを表示してください。

回答を表示

模範解答:

public class ClassComparison {
    public static void main(String[] args) {
        int[] classA = {85, 72, 90, 68, 95};
        int[] classB = {78, 88, 92, 65, 80};

        double avgA = calculateAverage(classA);
        double avgB = calculateAverage(classB);

        displayResult(avgA, avgB);
    }

    public static double calculateAverage(int[] scores) {
        int total = 0;
        for (int i = 0; i < scores.length; i++) {
            total += scores[i];
        }
        return (double) total / scores.length;
    }

    public static void displayResult(double avgA, double avgB) {
        System.out.println("クラスA平均点: " + avgA);
        System.out.println("クラスB平均点: " + avgB);

        if (avgA > avgB) {
            System.out.println("クラスAの方が高い");
        } else if (avgB > avgA) {
            System.out.println("クラスBの方が高い");
        } else {
            System.out.println("同じ平均点です");
        }
    }
}

ポイント: 計算処理と表示処理を別メソッドに分離することで、責務が明確になり再利用性が向上します。int型の除算で小数点を扱う場合は、(double)でキャストする必要があります。

メニュー制御
switch文を使い、以下のメニュー形式のプログラムを作成してください。
1: 合計点表示、2: 平均点表示、3: 最高点表示、4: 終了

回答を表示

模範解答:

public class MenuControl {
    public static void main(String[] args) {
        int[] scores = {85, 72, 90, 68, 95};
        int menu = 1; // 実際にはScannerで入力を受け取る

        processMenu(menu, scores);
    }

    public static void processMenu(int menu, int[] scores) {
        switch (menu) {
            case 1:
                System.out.println("合計点: " + calculateTotal(scores));
                break;
            case 2:
                System.out.println("平均点: " + calculateAverage(scores));
                break;
            case 3:
                System.out.println("最高点: " + findMax(scores));
                break;
            case 4:
                System.out.println("終了します");
                break;
            default:
                System.out.println("無効なメニューです");
        }
    }

    public static int calculateTotal(int[] scores) {
        int total = 0;
        for (int score : scores) {
            total += score;
        }
        return total;
    }

    public static double calculateAverage(int[] scores) {
        return (double) calculateTotal(scores) / scores.length;
    }

    public static int findMax(int[] scores) {
        int max = scores[0];
        for (int score : scores) {
            if (score > max) {
                max = score;
            }
        }
        return max;
    }
}

ポイント: switch文は複数の分岐を見やすく記述できます。各case文の最後には必ずbreakを入れます。拡張for文（for-each）を使うと配列の要素を簡潔に取得できます。

責務分離

public class RefactoredMenu {
    private static final int[] SCORES = {85, 72, 90, 68, 95};

    public static void main(String[] args) {
        // メニュー番号（実際にはScannerで入力）
        int menuNumber = 1;

        // メニュー処理を実行
        executeMenu(menuNumber);
    }

    private static void executeMenu(int menuNumber) {
        String result = getMenuResult(menuNumber);
        System.out.println(result);
    }

    private static String getMenuResult(int menuNumber) {
        switch (menuNumber) {
            case 1:
                return "合計点: " + ScoreCalculator.calculateTotal(SCORES);
            case 2:
                return "平均点: " + ScoreCalculator.calculateAverage(SCORES);
            case 3:
                return "最高点: " + ScoreCalculator.findMax(SCORES);
            case 4:
                return "終了します";
            default:
                return "無効なメニューです";
        }
    }
}

// 計算処理を専用クラスに分離
class ScoreCalculator {
    public static int calculateTotal(int[] scores) {
        int total = 0;
        for (int score : scores) {
            total += score;
        }
        return total;
    }

    public static double calculateAverage(int[] scores) {
        return (double) calculateTotal(scores) / scores.length;
    }

    public static int findMax(int[] scores) {
        int max = scores[0];
        for (int score : scores) {
            if (score > max) {
                max = score;
            }
        }
        return max;
    }
}

ポイント: mainメソッドは処理の起点として最小限にし、実際のロジックは別メソッド・別クラスに分離します。定数はstatic finalで定義し、クラス内で共有します。

/**
 * 成績処理メニューシステム
 * メニュー選択により各種統計情報を表示する
 */
public class ScoreMenuSystem {
    // 定数定義（メニュー番号）
    private static final int MENU_TOTAL = 1;
    private static final int MENU_AVERAGE = 2;
    private static final int MENU_MAX = 3;
    private static final int MENU_EXIT = 4;

    // テスト用成績データ
    private static final int[] TEST_SCORES = {85, 72, 90, 68, 95};

    public static void main(String[] args) {
        // ユーザーが選択したメニュー番号
        int selectedMenuNumber = MENU_TOTAL;

        // メニューに応じた処理を実行
        processSelectedMenu(selectedMenuNumber);
    }

    /**
     * 選択されたメニューに応じた処理を実行する
     * 処理結果を標準出力に表示
     */
    private static void processSelectedMenu(int menuNumber) {
        String resultMessage = generateResultMessage(menuNumber);
        System.out.println(resultMessage);
    }

    /**
     * メニュー番号に応じた結果メッセージを生成
     * 無効な番号の場合はエラーメッセージを返す
     */
    private static String generateResultMessage(int menuNumber) {
        switch (menuNumber) {
            case MENU_TOTAL:
                // 全体の合計点を計算して返す
                return "合計点: " + ScoreStatistics.calculateTotalScore(TEST_SCORES);
            case MENU_AVERAGE:
                // 平均点を小数点で返す（正確な評価のため）
                return "平均点: " + ScoreStatistics.calculateAverageScore(TEST_SCORES);
            case MENU_MAX:
                // 最も高い点数を返す
                return "最高点: " + ScoreStatistics.findMaximumScore(TEST_SCORES);
            case MENU_EXIT:
                return "終了します";
            default:
                return "無効なメニュー番号です";
        }
    }
}

/**
 * 成績統計計算クラス
 * 合計、平均、最大値などの統計処理を提供
 */
class ScoreStatistics {
    /**
     * 成績配列の合計点を計算
     */
    public static int calculateTotalScore(int[] scores) {
        int totalScore = 0;
        for (int individualScore : scores) {
            totalScore += individualScore;
        }
        return totalScore;
    }

    /**
     * 成績配列の平均点を計算
     * 小数点以下も含めて正確に計算する
     */
    public static double calculateAverageScore(int[] scores) {
        int total = calculateTotalScore(scores);
        // int除算では小数点が切り捨てられるため、doubleにキャスト
        return (double) total / scores.length;
    }

    /**
     * 成績配列から最高点を取得
     * 配列に要素が存在することを前提とする
     */
    public static int findMaximumScore(int[] scores) {
        int maximumScore = scores[0]; // 最初の要素で初期化
        for (int currentScore : scores) {
            // より高い点数があれば更新
            if (currentScore > maximumScore) {
                maximumScore = currentScore;
            }
        }
        return maximumScore;
    }
}

ポイント: 変数名・メソッド名は動作が一目でわかるように具体的に命名します。コメントは「何をするか」ではなく「なぜそうするか」を記述します。メニュー番号のような定数は必ず定数化します。
