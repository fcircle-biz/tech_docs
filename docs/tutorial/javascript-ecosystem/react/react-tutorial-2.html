2.1 Docker ComposeによるPostgreSQL環境構築

PostgreSQLとpgAdminをDockerコンテナで起動します。これにより、環境を簡単に構築・破棄できます。

実装 2-1: Docker Compose設定ファイル作成
手順
  1. プロジェクトルートに戻る: 
    # フロントエンドディレクトリから親ディレクトリに移動
cd ..

# 新しいプロジェクトフォルダを作成（既に親ディレクトリにいる場合）
mkdir react-user-management
cd react-user-management

# 既存のフロントエンドプロジェクトを移動
mv ../user-management-frontend ./frontend
  2. docker-compose.ymlファイルを作成: 
    version: '3.8'

services:
  postgres:
    image: postgres:16-alpine
    container_name: react-postgres
    environment:
      POSTGRES_USER: reactuser
      POSTGRES_PASSWORD: reactpass
      POSTGRES_DB: userdb
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    restart: unless-stopped

  pgadmin:
    image: dpage/pgadmin4:latest
    container_name: react-pgadmin
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@example.com
      PGADMIN_DEFAULT_PASSWORD: admin
    ports:
      - "5050:80"
    depends_on:
      - postgres
    restart: unless-stopped

volumes:
  postgres_data:
  3. 初期SQLファイル（init.sql）を作成: 
    -- ユーザーテーブルの作成
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    birth_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- インデックスの作成
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_email ON users(email);

-- サンプルデータの挿入（パスワードは'password123'のハッシュ）
INSERT INTO users (username, email, password_hash, full_name, birth_date) VALUES
('admin', 'admin@example.com', '$2b$10$abcdefghijklmnopqrstuvwxyz1234567890ABCDEFGHIJ', '管理者', '1990-01-01'),
('tanaka', 'tanaka@example.com', '$2b$10$abcdefghijklmnopqrstuvwxyz1234567890ABCDEFGHIJ', '田中太郎', '1985-12-20'),
('yamada', 'yamada@example.com', '$2b$10$abcdefghijklmnopqrstuvwxyz1234567890ABCDEFGHIJ', '山田花子', '1992-03-08');

    docker-compose up -d

2.4 PostgreSQL接続設定

実装 2-4: データベース接続の作成
手順
  1. .envファイルを作成: 
    # データベース接続
DATABASE_URL=postgresql://reactuser:reactpass@localhost:5432/userdb
DB_HOST=localhost
DB_PORT=5432
DB_USER=reactuser
DB_PASSWORD=reactpass
DB_NAME=userdb

# サーバー設定
PORT=3000
NODE_ENV=development

# JWT設定
JWT_SECRET=your-secret-key-change-in-production
JWT_EXPIRES_IN=24h
  2. src/config/database.jsを作成: 
    import pkg from 'pg';
const { Pool } = pkg;
import dotenv from 'dotenv';

dotenv.config();

const pool = new Pool({
  host: process.env.DB_HOST,
  port: process.env.DB_PORT,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_NAME,
});

// 接続テスト
pool.on('connect', () => {
  console.log('データベースに接続しました');
});

pool.on('error', (err) => {
  console.error('データベース接続エラー:', err);
  process.exit(-1);
});

export default pool;

2.5 Express APIサーバーの基本構造

実装 2-5: APIサーバーの作成
手順
  1. src/app.jsを作成: 
    import express from 'express';
import cors from 'cors';
import dotenv from 'dotenv';

dotenv.config();

const app = express();

// ミドルウェア
app.use(cors({
  origin: 'http://localhost:5173', // Viteのデフォルトポート
  credentials: true,
}));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// ルート
app.get('/', (req, res) => {
  res.json({ message: 'User Management API' });
});

// エラーハンドリングミドルウェア
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({
    success: false,
    message: 'サーバーエラーが発生しました',
    error: process.env.NODE_ENV === 'development' ? err.message : {},
  });
});

export default app;
  2. src/server.jsを作成: 
    import app from './app.js';
import dotenv from 'dotenv';

dotenv.config();

const PORT = process.env.PORT || 3000;

app.listen(PORT, () => {
  console.log(`サーバーがポート ${PORT} で起動しました`);
  console.log(`環境: ${process.env.NODE_ENV}`);
});

npm run dev

2.6 ユーザーAPIエンドポイントの実装

実装 2-6: ユーザーコントローラーとルート作成
手順
  1. src/controllers/userController.jsを作成: 
    import db from '../config/database.js';

// 全ユーザー取得
export const getAllUsers = async (req, res) => {
  try {
    const { page = 1, limit = 10 } = req.query;
    const offset = (page - 1) * limit;

    const result = await db.query(
      'SELECT id, username, email, full_name, birth_date, created_at FROM users ORDER BY id LIMIT $1 OFFSET $2',
      [limit, offset]
    );

    const countResult = await db.query('SELECT COUNT(*) FROM users');
    const total = parseInt(countResult.rows[0].count);

    res.json({
      success: true,
      data: result.rows,
      pagination: {
        page: parseInt(page),
        limit: parseInt(limit),
        total,
        totalPages: Math.ceil(total / limit),
      },
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      message: 'ユーザー取得エラー',
      error: error.message,
    });
  }
};

// 特定ユーザー取得
export const getUserById = async (req, res) => {
  try {
    const { id } = req.params;
    const result = await db.query(
      'SELECT id, username, email, full_name, birth_date, created_at FROM users WHERE id = $1',
      [id]
    );

    if (result.rows.length === 0) {
      return res.status(404).json({
        success: false,
        message: 'ユーザーが見つかりません',
      });
    }

    res.json({
      success: true,
      data: result.rows[0],
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      message: 'ユーザー取得エラー',
      error: error.message,
    });
  }
};
  2. src/routes/userRoutes.jsを作成: 
    import express from 'express';
import { getAllUsers, getUserById } from '../controllers/userController.js';

const router = express.Router();

router.get('/', getAllUsers);
router.get('/:id', getUserById);

export default router;
  3. src/app.jsにルートを追加: 
    import express from 'express';
import cors from 'cors';
import dotenv from 'dotenv';
import userRoutes from './routes/userRoutes.js';

dotenv.config();

const app = express();

// ミドルウェア
app.use(cors({
  origin: 'http://localhost:5173',
  credentials: true,
}));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// ルート
app.get('/', (req, res) => {
  res.json({ message: 'User Management API' });
});

app.use('/api/users', userRoutes);

// エラーハンドリングミドルウェア
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({
    success: false,
    message: 'サーバーエラーが発生しました',
    error: process.env.NODE_ENV === 'development' ? err.message : {},
  });
});

export default app;
