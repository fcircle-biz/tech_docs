8.1 Zodによる高度なバリデーション

実装 8-1: カスタムバリデーションスキーマ
// src/lib/validations.ts を拡張
import { z } from 'zod';

const passwordRegex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/;

export const userSchema = z.object({
  name: z.string()
    .min(2, 'お名前は2文字以上')
    .max(50, 'お名前は50文字以下'),
  email: z.string()
    .email('正しいメールアドレスを入力')
    .refine(
      async (email) => {
        const user = await prisma.user.findUnique({
          where: { email },
        });
        return !user; // 既存ユーザーの場合はfalse
      },
      { message: 'このメールアドレスは既に登録されています' }
    ),
  password: z.string()
    .min(8, 'パスワードは8文字以上')
    .regex(passwordRegex, '大文字・小文字・数字を含めてください'),
}).refine((data) => data.password === data.confirmPassword, {
  message: 'パスワードが一致しません',
  path: ['confirmPassword'],
});

8.2 リアルタイムバリデーション

Client Sideバリデーションの例
// 入力時にバリデーション
'use client';

import { userSchema } from '@/lib/validations';
import { useState } from 'react';

export default function ValidatingForm() {
  const [errors, setErrors] = useState<Record<string, string>>({});

  const handleChange = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;

    try {
      // 該当フィールドのバリデーション
      await userSchema.pick({ [name]: true }).parseAsync({
        [name]: value,
      });
      // バリデーション成功時はエラーをクリア
      setErrors((prev) => {
        const newErrors = { ...prev };
        delete newErrors[name];
        return newErrors;
      });
    } catch (error) {
      if (error instanceof z.ZodError) {
        const fieldError = error.errors[0];
        setErrors((prev) => ({
          ...prev,
          [name]: fieldError.message,
        }));
      }
    }
  };

  return (
    <div>
      <input name="email" onChange={handleChange} />
      {errors.email && <p className="text-red-600">{errors.email}</p>}
    </div>
  );
}

8.4 トーストメッセージ通知

トーストメッセージコンポーネントの例
// src/app/components/toast.tsx
'use client';

import { useState, useEffect } from 'react';

interface ToastProps {
  message: string;
  type: 'success' | 'error' | 'info';
  duration?: number;
}

export function useToast() {
  const [toast, setToast] = useState<ToastProps | null>(null);

  const showToast = (message: string, type: ToastProps['type'], duration = 3000) => {
    setToast({ message, type, duration });
    setTimeout(() => setToast(null), duration);
  };

  return { toast, showToast };
}

export default function Toast({ message, type, duration = 3000 }: ToastProps) {
  const [isVisible, setIsVisible] = useState(true);

  useEffect(() => {
    const timer = setTimeout(() => setIsVisible(false), duration);
    return () => clearTimeout(timer);
  }, [duration]);

  if (!isVisible) return null;

  const bgColor = {
    success: 'bg-green-100 text-green-800',
    error: 'bg-red-100 text-red-800',
    info: 'bg-blue-100 text-blue-800',
  }[type];

  return (
    <div className={`fixed bottom-4 right-4 p-4 rounded ${bgColor}`}>
      {message}
    </div>
  );
}

8.5 API エラーハンドリング

Server Action のエラー処理パターン
// src/actions/user-actions.ts にエラーハンドリングを追加
export async function createUser(formData: FormData) {
  try {
    // バリデーション
    const data = userSchema.parse({...});

    // データベース操作
    const user = await prisma.user.create({ data });

    revalidatePath('/users');

    return { success: true, user };
  } catch (error) {
    if (error instanceof z.ZodError) {
      return {
        success: false,
        errors: error.flatten().fieldErrors,
      };
    }

    if (error instanceof PrismaClientKnownRequestError) {
      if (error.code === 'P2002') {
        return {
          success: false,
          error: '既に登録されているメールアドレスです',
        };
      }
    }

    console.error('[Database Error]', error);
    return {
      success: false,
      error: 'サーバーエラーが発生しました',
    };
  }
}
