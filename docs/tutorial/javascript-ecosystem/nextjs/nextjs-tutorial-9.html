9.1 Jestのセットアップ

実装 9-1: テスト環境の構築
# テスティングライブラリをインストール
npm install --save-dev jest @testing-library/react @testing-library/jest-dom jest-environment-jsdom

# Jest設定ファイルを作成
npx jest --init
jest.config.js の設定例
const nextJest = require('next/jest')

const createJestConfig = nextJest({
  dir: './',
})

const customJestConfig = {
  setupFilesAfterEnv: ['<rootDir>/jest.setup.js'],
  testEnvironment: 'jest-environment-jsdom',
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
  },
}

module.exports = createJestConfig(customJestConfig)

9.2 コンポーネントテスト

React Testing Libraryでのテスト例
// src/__tests__/components/register-form.test.tsx
import { render, screen, fireEvent } from '@testing-library/react';
import RegisterForm from '@/app/components/register-form';

describe('RegisterForm', () => {
  it('フォームが正常にレンダリングされる', () => {
    render(<RegisterForm />);
    expect(screen.getByLabelText('お名前')).toBeInTheDocument();
  });

  it('入力値が更新される', () => {
    render(<RegisterForm />);
    const nameInput = screen.getByLabelText('お名前') as HTMLInputElement;
    fireEvent.change(nameInput, { target: { value: 'テストユーザー' } });
    expect(nameInput.value).toBe('テストユーザー');
  });

  it('バリデーションエラーが表示される', async () => {
    render(<RegisterForm />);
    const submitButton = screen.getByText('ユーザー登録');
    fireEvent.click(submitButton);

    const error = await screen.findByText(/2文字以上/i);
    expect(error).toBeInTheDocument();
  });
});

9.3 Server Actionsのテスト

Server Action のユニットテスト
// src/__tests__/actions/user-actions.test.ts
import { registerUser } from '@/actions/user-actions';
import { prisma } from '@/lib/prisma';

jest.mock('@/lib/prisma', () => ({
  prisma: {
    user: {
      create: jest.fn(),
      findUnique: jest.fn(),
    },
  },
}));

describe('User Actions', () => {
  it('新規ユーザーを登録できる', async () => {
    const formData = new FormData();
    formData.append('name', 'テストユーザー');
    formData.append('email', 'test@example.com');
    formData.append('password', 'SecurePass123');
    formData.append('confirmPassword', 'SecurePass123');

    (prisma.user.create as jest.Mock).mockResolvedValue({
      id: '1',
      name: 'テストユーザー',
      email: 'test@example.com',
    });

    const result = await registerUser(formData);
    expect(result.success).toBe(true);
  });

  it('既存メールアドレスで登録できない', async () => {
    const formData = new FormData();
    formData.append('email', 'existing@example.com');

    (prisma.user.findUnique as jest.Mock).mockResolvedValue({
      id: '1',
      email: 'existing@example.com',
    });

    const result = await registerUser(formData);
    expect(result.success).toBe(false);
  });
});

9.4 Playwrightによるエンドツーエンドテスト

Playwright セットアップ
# Playwright をインストール
npm install --save-dev @playwright/test

# playwright.config.ts を作成
npx playwright install
E2E テストの例
// e2e/user-flow.spec.ts
import { test, expect } from '@playwright/test';

test.describe('User Management Flow', () => {
  test('ユーザー登録から一覧表示までのフロー', async ({ page }) => {
    // ホームページにアクセス
    await page.goto('http://localhost:3000');
    expect(page).toHaveTitle('ユーザー管理システム');

    // 登録ページに遷移
    await page.click('text=新規登録');

    // フォーム入力
    await page.fill('input[name="name"]', 'テストユーザー');
    await page.fill('input[name="email"]', 'test@example.com');
    await page.fill('input[name="password"]', 'SecurePass123');

    // 送信
    await page.click('button:has-text("登録")');

    // ユーザー一覧ページに遷移することを確認
    await expect(page).toHaveURL('http://localhost:3000/users');
  });
});
