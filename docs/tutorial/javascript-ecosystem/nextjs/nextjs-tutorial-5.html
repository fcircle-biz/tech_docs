5.1 ユーザー一覧ページの実装

ユーザーデータをデータベースから取得し、テーブル形式で表示します。

実装 5-1: ユーザー一覧ページ

src/app/users/page.tsx を更新します。

手順
  1. ユーザー一覧ページを実装: 
    // src/app/users/page.tsx
import { prisma } from '@/lib/prisma';
import Link from 'next/link';
import { Metadata } from 'next';

export const metadata: Metadata = {
  title: 'ユーザー一覧',
};

export default async function UsersPage() {
  // すべてのユーザーを取得
  const users = await prisma.user.findMany({
    orderBy: { createdAt: 'desc' },
  });

  return (
    <div className="container mx-auto py-10">
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-3xl font-bold">ユーザー一覧</h1>
        <Link
          href="/users/new"
          className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
        >
          新規登録
        </Link>
      </div>

      {users.length === 0 ? (
        <div className="text-center py-10">
          <p className="text-gray-600">ユーザーが登録されていません</p>
        </div>
      ) : (
        <div className="overflow-x-auto">
          <table className="min-w-full border-collapse">
            <thead>
              <tr className="bg-gray-100">
                <th className="border p-2 text-left">ID</th>
                <th className="border p-2 text-left">名前</th>
                <th className="border p-2 text-left">メールアドレス</th>
                <th className="border p-2 text-left">作成日</th>
                <th className="border p-2 text-left">アクション</th>
              </tr>
            </thead>
            <tbody>
              {users.map((user) => (
                <tr key={user.id} className="hover:bg-gray-50">
                  <td className="border p-2">{user.id}</td>
                  <td className="border p-2">{user.name}</td>
                  <td className="border p-2">{user.email}</td>
                  <td className="border p-2">
                    {user.createdAt.toLocaleDateString('ja-JP')}
                  </td>
                  <td className="border p-2">
                    <Link
                      href={`/users/${user.id}`}
                      className="text-blue-500 hover:underline"
                    >
                      詳細
                    </Link>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
}
期待される結果

/users にアクセスすると、登録済みのユーザーがテーブル形式で表示されます。

5.2 ユーザー詳細ページの実装

動的ルートを使用して、ユーザーの詳細情報を表示します。

実装 5-2: ユーザー詳細ページ

src/app/users/[id]/page.tsx を更新します。

手順
  1. 詳細ページを実装: 
    // src/app/users/[id]/page.tsx
import { prisma } from '@/lib/prisma';
import Link from 'next/link';
import { notFound } from 'next/navigation';

type Props = {
  params: { id: string };
};

export async function generateMetadata({ params }: Props) {
  const user = await prisma.user.findUnique({
    where: { id: params.id },
  });
  return {
    title: user ? user.name : 'ユーザーが見つかりません',
  };
}

export default async function UserDetailPage({ params }: Props) {
  const user = await prisma.user.findUnique({
    where: { id: params.id },
  });

  if (!user) {
    notFound();
  }

  return (
    <div className="container mx-auto py-10">
      <Link
        href="/users"
        className="text-blue-500 hover:underline mb-4 inline-block"
      >
        ← 一覧に戻る
      </Link>

      <div className="bg-white border rounded p-6">
        <h1 className="text-3xl font-bold mb-4">{user.name}</h1>
        <div className="space-y-2">
          <p><strong>ID:</strong> {user.id}</p>
          <p><strong>メール:</strong> {user.email}</p>
          <p><strong>作成日:</strong> {user.createdAt.toLocaleString('ja-JP')}</p>
          <p><strong>更新日:</strong> {user.updatedAt.toLocaleString('ja-JP')}</p>
        </div>

        <div className="mt-6 flex gap-2">
          <Link
            href={`/users/${user.id}/edit`}
            className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
          >
            編集
          </Link>
        </div>
      </div>
    </div>
  );
}
期待される結果

/users/[id] にアクセスすると、そのユーザーの詳細情報が表示されます。

// src/app/users/page.tsx に検索パラメータを追加
import { Suspense } from 'react';

type SearchParams = {
  q?: string;
};

async function UsersList({ searchParams }: { searchParams: SearchParams }) {
  const query = searchParams.q || '';

  let users = await prisma.user.findMany({
    where: query
      ? {
          OR: [
            { name: { contains: query, mode: 'insensitive' } },
            { email: { contains: query, mode: 'insensitive' } },
          ],
        }
      : {},
    orderBy: { createdAt: 'desc' },
  });

  // テーブル出力...
}

export default function UsersPage({
  searchParams,
}: {
  searchParams: SearchParams;
}) {
  return (
    <div className="container mx-auto py-10">
      <form className="mb-6">
        <input
          type="text"
          name="q"
          placeholder="名前またはメールで検索..."
          defaultValue={searchParams.q}
          className="w-full px-4 py-2 border rounded"
        />
      </form>
      <Suspense fallback={<div>読み込み中...</div>}>
        <UsersList searchParams={searchParams} />
      </Suspense>
    </div>
  );
}
