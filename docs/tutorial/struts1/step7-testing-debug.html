TestDatabaseManager.java

src/test/java/com/example/struts/db/TestDatabaseManager.java:

package com.example.struts.db;

import java.io.IOException;
import java.io.InputStream;
import java.sql.Connection;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.Properties;

import org.apache.commons.dbcp2.BasicDataSource;

public class TestDatabaseManager {
    
    private static TestDatabaseManager instance;
    private BasicDataSource dataSource;
    
    private TestDatabaseManager() {
        initializeDataSource();
        initializeSchema();
    }
    
    public static synchronized TestDatabaseManager getInstance() {
        if (instance == null) {
            instance = new TestDatabaseManager();
        }
        return instance;
    }
    
    private void initializeDataSource() {
        try {
            Properties props = new Properties();
            InputStream is = getClass().getClassLoader()
                .getResourceAsStream("database-test.properties");
            props.load(is);
            is.close();
            
            dataSource = new BasicDataSource();
            dataSource.setDriverClassName(props.getProperty("db.driver"));
            dataSource.setUrl(props.getProperty("db.url"));
            dataSource.setUsername(props.getProperty("db.username"));
            dataSource.setPassword(props.getProperty("db.password"));
            dataSource.setInitialSize(Integer.parseInt(props.getProperty("db.initialSize", "2")));
            dataSource.setMaxTotal(Integer.parseInt(props.getProperty("db.maxTotal", "5")));
            
        } catch (IOException e) {
            throw new RuntimeException("テスト用データベース設定の読み込みに失敗しました", e);
        }
    }
    
    private void initializeSchema() {
        try (Connection conn = dataSource.getConnection();
             Statement stmt = conn.createStatement()) {
            
            // テーブル作成
            String createTableSql = 
                "CREATE TABLE IF NOT EXISTS users (" +
                "user_id IDENTITY PRIMARY KEY, " +
                "username VARCHAR(50) NOT NULL UNIQUE, " +
                "email VARCHAR(100) NOT NULL, " +
                "full_name VARCHAR(100) NOT NULL, " +
                "birth_date DATE, " +
                "created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, " +
                "updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP" +
                ")";
            
            stmt.execute(createTableSql);
            
        } catch (SQLException e) {
            throw new RuntimeException("テストスキーマの初期化に失敗しました", e);
        }
    }
    
    public Connection getConnection() throws SQLException {
        return dataSource.getConnection();
    }
    
    public void clearData() {
        try (Connection conn = dataSource.getConnection();
             Statement stmt = conn.createStatement()) {
            
            stmt.execute("DELETE FROM users");
            stmt.execute("ALTER SEQUENCE PUBLIC.SYSTEM_SEQUENCE_1 RESTART WITH 1");
            
        } catch (SQLException e) {
            throw new RuntimeException("テストデータのクリアに失敗しました", e);
        }
    }
    
    public void shutdown() throws SQLException {
        if (dataSource != null) {
            dataSource.close();
        }
    }
}

実習 7-2: UserCreateActionテスト実装

UserCreateActionのテストクラスを作成します。

UserCreateActionTest.java

src/test/java/com/example/struts/action/UserCreateActionTest.java:

package com.example.struts.action;

import static org.junit.Assert.*;
import static org.mockito.Mockito.*;

import java.sql.Date;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import org.apache.struts.action.ActionForward;
import org.apache.struts.action.ActionMapping;
import org.apache.struts.action.ActionMessages;
import org.junit.Before;
import org.junit.Test;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

import com.example.struts.form.UserForm;
import com.example.struts.model.User;
import com.example.struts.service.UserService;

public class UserCreateActionTest {
    
    @Mock private ActionMapping mockMapping;
    @Mock private HttpServletRequest mockRequest;
    @Mock private HttpServletResponse mockResponse;
    @Mock private UserService mockUserService;
    
    private UserCreateAction action;
    private UserForm userForm;
    
    @Before
    public void setUp() {
        MockitoAnnotations.initMocks(this);
        
        action = new UserCreateAction() {
            // UserServiceをモックでオーバーライド
            private UserService getUserService() {
                return mockUserService;
            }
        };
        
        userForm = new UserForm();
        userForm.setUsername("testuser");
        userForm.setEmail("test@example.com");
        userForm.setFullName("テストユーザー");
        userForm.setBirthDate("1990-01-01");
    }
    
    @Test
    public void testExecute_Success() throws Exception {
        // モック設定
        User createdUser = new User("testuser", "test@example.com", "テストユーザー", 
                                   Date.valueOf("1990-01-01"));
        createdUser.setUserId(1);
        
        when(mockUserService.createUser(any(User.class))).thenReturn(createdUser);
        
        ActionForward expectedForward = new ActionForward();
        when(mockMapping.findForward("success")).thenReturn(expectedForward);
        
        // テスト実行
        ActionForward result = action.execute(mockMapping, userForm, 
                                            mockRequest, mockResponse);
        
        // 検証
        assertEquals(expectedForward, result);
        verify(mockUserService).createUser(any(User.class));
        
        // メッセージが保存されることを確認
        verify(mockRequest).setAttribute(eq("org.apache.struts.action.ACTION_MESSAGE"), 
                                       any(ActionMessages.class));
    }
    
    @Test
    public void testExecute_DuplicateUser() throws Exception {
        // モック設定 - 重複ユーザーエラー
        when(mockUserService.createUser(any(User.class)))
            .thenThrow(new Exception("ユーザー名が既に存在します"));
        
        ActionForward inputForward = new ActionForward();
        when(mockMapping.getInputForward()).thenReturn(inputForward);
        
        // テスト実行
        ActionForward result = action.execute(mockMapping, userForm, 
                                            mockRequest, mockResponse);
        
        // 検証
        assertEquals(inputForward, result);
        verify(mockRequest).setAttribute(eq("org.apache.struts.action.ACTION_ERROR"), 
                                       any(ActionMessages.class));
    }
    
    @Test
    public void testExecute_InvalidBirthDate() throws Exception {
        // 不正な日付フォーマットを設定
        userForm.setBirthDate("invalid-date");
        
        ActionForward inputForward = new ActionForward();
        when(mockMapping.getInputForward()).thenReturn(inputForward);
        
        // テスト実行
        ActionForward result = action.execute(mockMapping, userForm, 
                                            mockRequest, mockResponse);
        
        // 検証
        assertEquals(inputForward, result);
        verify(mockRequest).setAttribute(eq("org.apache.struts.action.ACTION_ERROR"), 
                                       any(ActionMessages.class));
        
        // UserServiceが呼ばれていないことを確認
        verify(mockUserService, never()).createUser(any(User.class));
    }
    
    @Test
    public void testExecute_SystemError() throws Exception {
        // システムエラーのシミュレーション
        when(mockUserService.createUser(any(User.class)))
            .thenThrow(new RuntimeException("データベースエラー"));
        
        ActionForward inputForward = new ActionForward();
        when(mockMapping.getInputForward()).thenReturn(inputForward);
        
        // テスト実行
        ActionForward result = action.execute(mockMapping, userForm, 
                                            mockRequest, mockResponse);
        
        // 検証
        assertEquals(inputForward, result);
        verify(mockRequest).setAttribute(eq("org.apache.struts.action.ACTION_ERROR"), 
                                       any(ActionMessages.class));
    }
}

package com.example.struts.dao.impl;

import static org.junit.Assert.*;

import java.sql.Date;
import java.util.List;

import org.junit.After;
import org.junit.Before;
import org.junit.Test;

import com.example.struts.dao.UserDAO;
import com.example.struts.db.TestDatabaseManager;
import com.example.struts.model.User;

public class UserDAOImplTest {
    
    private UserDAO userDAO;
    private TestDatabaseManager testDbManager;
    
    @Before
    public void setUp() {
        testDbManager = TestDatabaseManager.getInstance();
        userDAO = new UserDAOImpl() {
            @Override
            protected Connection getConnection() throws SQLException {
                return testDbManager.getConnection();
            }
        };
        
        // テストデータをクリア
        testDbManager.clearData();
    }
    
    @After
    public void tearDown() {
        testDbManager.clearData();
    }
    
    @Test
    public void testInsert_Success() throws Exception {
        // テストデータ準備
        User user = new User("testuser", "test@example.com", "テストユーザー", 
                           Date.valueOf("1990-01-01"));
        
        // テスト実行
        User result = userDAO.insert(user);
        
        // 検証
        assertNotNull(result);
        assertTrue(result.getUserId() > 0);
        assertEquals("testuser", result.getUsername());
        assertEquals("test@example.com", result.getEmail());
        assertEquals("テストユーザー", result.getFullName());
    }
    
    @Test
    public void testFindById_Found() throws Exception {
        // テストデータ作成
        User user = new User("testuser", "test@example.com", "テストユーザー", 
                           Date.valueOf("1990-01-01"));
        User insertedUser = userDAO.insert(user);
        
        // テスト実行
        User result = userDAO.findById(insertedUser.getUserId());
        
        // 検証
        assertNotNull(result);
        assertEquals(insertedUser.getUserId(), result.getUserId());
        assertEquals("testuser", result.getUsername());
    }
    
    @Test
    public void testFindById_NotFound() throws Exception {
        // テスト実行
        User result = userDAO.findById(999);

                           Date.valueOf("1990-01-01"));
        userDAO.insert(user);
        
        // テスト実行
        User result = userDAO.findByUsername("testuser");
        
        // 検証
        assertNotNull(result);
        assertEquals("testuser", result.getUsername());
    }
    
    @Test
    public void testFindByUsername_NotFound() throws Exception {
        // テスト実行
        User result = userDAO.findByUsername("nonexistent");
        
        // 検証
        assertNull(result);
    }
    
    @Test
    public void testFindAll_Empty() throws Exception {
        // テスト実行
        List<User> result = userDAO.findAll();
        
        // 検証
        assertNotNull(result);
        assertEquals(0, result.size());
    }
    
    @Test
    public void testFindAll_MultipleUsers() throws Exception {
        // テストデータ作成
        User user1 = new User("user1", "user1@example.com", "ユーザー1", 
                            Date.valueOf("1990-01-01"));
        User user2 = new User("user2", "user2@example.com", "ユーザー2", 
                            Date.valueOf("1991-01-01"));
        
        userDAO.insert(user1);
        userDAO.insert(user2);
        
        // テスト実行
        List<User> result = userDAO.findAll();
        
        // 検証
        assertNotNull(result);
        assertEquals(2, result.size());
    }
    
    @Test
    public void testUpdate_Success() throws Exception {
        // テストデータ作成
        User user = new User("testuser", "test@example.com", "テストユーザー", 
                           Date.valueOf("1990-01-01"));
        User insertedUser = userDAO.insert(user);
        
        // 更新データ設定
        insertedUser.setEmail("updated@example.com");
        insertedUser.setFullName("更新されたユーザー");
        
        // テスト実行
        boolean result = userDAO.update(insertedUser);
        
        // 検証
        assertTrue(result);
        
        User updatedUser = userDAO.findById(insertedUser.getUserId());
        assertEquals("updated@example.com", updatedUser.getEmail());
        assertEquals("更新されたユーザー", updatedUser.getFullName());
    }
    
    @Test
    public void testUpdate_NotFound() throws Exception {

        // テスト実行
        boolean result = userDAO.update(user);
        
        // 検証
        assertFalse(result);
    }
    
    @Test
    public void testDelete_Success() throws Exception {
        // テストデータ作成
        User user = new User("testuser", "test@example.com", "テストユーザー", 
                           Date.valueOf("1990-01-01"));
        User insertedUser = userDAO.insert(user);
        
        // テスト実行
        boolean result = userDAO.delete(insertedUser.getUserId());
        
        // 検証
        assertTrue(result);
        
        User deletedUser = userDAO.findById(insertedUser.getUserId());
        assertNull(deletedUser);
    }
    
    @Test
    public void testDelete_NotFound() throws Exception {
        // テスト実行
        boolean result = userDAO.delete(999);
        
        // 検証
        assertFalse(result);
    }
    
    @Test
    public void testCount_Empty() throws Exception {
        // テスト実行
        int result = userDAO.count();
        
        // 検証

    }
    
    @Test

                            Date.valueOf("1990-01-01"));
        User user2 = new User("user2", "user2@example.com", "ユーザー2",

        userDAO.insert(user1);
        userDAO.insert(user2);
        
        // テスト実行
        int result = userDAO.count();
        
        // 検証
        assertEquals(2, result);
    }
}

// Selenium WebDriver を使用したブラウザテスト例
const {Builder, By, Key, until} = require('selenium-webdriver');

async function testUserRegistration() {
    let driver = await new Builder().forBrowser('chrome').build();
    
    try {
        // ユーザー登録画面にアクセス
        await driver.get('http://localhost:8080/struts-tutorial/userForm.do');
        
        // フォーム入力
        await driver.findElement(By.id('username')).sendKeys('seleniumuser');
        await driver.findElement(By.id('email')).sendKeys('selenium@example.com');
        await driver.findElement(By.id('fullName')).sendKeys('Seleniumユーザー');
        await driver.findElement(By.id('birthDate')).sendKeys('1990-01-01');
        
        // 送信
        await driver.findElement(By.css('input[type="submit"]')).click();
        
        // 結果確認
        await driver.wait(until.titleContains('ユーザー一覧'), 5000);
        
        const successMessage = await driver.findElement(By.css('.message'));
        const messageText = await successMessage.getText();
        
        if (messageText.includes('seleniumuser')) {
            console.log('ユーザー登録テスト: 成功');
        } else {
            console.log('ユーザー登録テスト: 失敗');
        }
        
    } finally {
        await driver.quit();
    }
}

LogUtil.java

src/main/java/com/example/struts/util/LogUtil.java:

package com.example.struts.util;

import org.apache.log4j.Logger;

public class LogUtil {
    
    /**
     * クラス用のLoggerを取得
     */
    public static Logger getLogger(Class<?> clazz) {
        return Logger.getLogger(clazz);
    }
    
    /**
     * メソッド開始ログ出力
     */
    public static void logMethodStart(Logger logger, String methodName, Object... params) {
        if (logger.isDebugEnabled()) {
            StringBuilder sb = new StringBuilder();
            sb.append("START - ").append(methodName).append("(");
            
            if (params != null && params.length > 0) {
                for (int i = 0; i < params.length; i++) {
                    if (i > 0) sb.append(", ");
                    sb.append(params[i]);
                }
            }
            
            sb.append(")");
            logger.debug(sb.toString());
        }
    }
    
    /**
     * メソッド終了ログ出力
     */
    public static void logMethodEnd(Logger logger, String methodName, Object result) {
        if (logger.isDebugEnabled()) {
            logger.debug("END - " + methodName + " : " + result);
        }
    }
    
    /**
     * SQLログ出力
     */
    public static void logSQL(Logger logger, String sql, Object... params) {
        if (logger.isDebugEnabled()) {
            StringBuilder sb = new StringBuilder();
            sb.append("SQL: ").append(sql);
            
            if (params != null && params.length > 0) {
                sb.append(" | Parameters: ");
                for (int i = 0; i < params.length; i++) {
                    if (i > 0) sb.append(", ");
                    sb.append("[").append(i+1).append("]=").append(params[i]);
                }
            }
            
            logger.debug(sb.toString());
        }
    }
    
    /**
     * エラーログ出力（スタックトレース付き）
     */
    public static void logError(Logger logger, String message, Throwable throwable) {
        logger.error(message, throwable);
    }
    
    /**
     * パフォーマンス測定用
     */
    public static class PerformanceTimer {
        private long startTime;
        private String operation;
        private Logger logger;
        
        public PerformanceTimer(Logger logger, String operation) {
            this.logger = logger;
            this.operation = operation;
            this.startTime = System.currentTimeMillis();
            
            if (logger.isDebugEnabled()) {
                logger.debug("PERF START - " + operation);
            }
        }
        
        public void stop() {
            long duration = System.currentTimeMillis() - startTime;
            
            if (logger.isDebugEnabled()) {
                logger.debug("PERF END - " + operation + " : " + duration + "ms");
            }
            
            if (duration > 1000) {
                logger.warn("SLOW OPERATION - " + operation + " took " + duration + "ms");
            }
        }
    }
}

IDEデバッグ設定（Eclipse例）
Eclipseでのデバッグ設定手順:

1. プロジェクトを右クリック → Debug As → Debug on Server
2. ブレークポイントを設定:
   - Action クラスのexecute()メソッド
   - DAO クラスのCRUD操作メソッド
   - JSP の表示箇所

3. デバッグ実行中の確認項目:
   - 変数の値
   - リクエストパラメータ
   - セッション情報
   - データベースの状態

4. ステップ実行でフローを確認:
   - Step Over (F6): 次の行へ
   - Step Into (F5): メソッド内部へ
   - Step Out (F7): メソッドから抜ける
⚠️ テストとデバッグのベストプラクティス
  • テストは開発と並行して作成し、継続的に実行してください
  • 単体テスト、統合テスト、システムテストを段階的に実施してください
  • 本番環境では適切なログレベルを設定してください（INFOまたはWARN）
  • 機密情報（パスワード等）はログに出力しないでください
  • パフォーマンステストも定期的に実施してください
  • テストデータは本番データと分離して管理してください
