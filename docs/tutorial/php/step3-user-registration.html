<?php
require_once '../config/database.php';

// エラーメッセージとフォームデータの初期化
$errors = [];
$form_data = [
    'username' => '',
    'email' => '',
    'full_name' => '',
    'birth_date' => ''
];
$success_message = '';

// POST送信時の処理
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // フォームデータの取得とサニタイズ
    $form_data['username'] = trim($_POST['username'] ?? '');
    $form_data['email'] = trim($_POST['email'] ?? '');
    $form_data['full_name'] = trim($_POST['full_name'] ?? '');
    $form_data['birth_date'] = trim($_POST['birth_date'] ?? '');
    
    // バリデーション実行
    $errors = validateUserData($form_data);
    
    // エラーがない場合、データベースに挿入
    if (empty($errors)) {
        try {
            $pdo = getDatabase();
            
            // 重複チェック
            if (isUserExists($pdo, $form_data['username'], $form_data['email'])) {
                $errors['duplicate'] = 'ユーザー名またはメールアドレスが既に使用されています。';
            } else {
                // ユーザー挿入
                if (insertUser($pdo, $form_data)) {
                    $success_message = "ユーザー「{$form_data['full_name']}」を正常に登録しました。";
                    // フォームデータをリセット
                    $form_data = ['username' => '', 'email' => '', 'full_name' => '', 'birth_date' => ''];
                }
            }
        } catch (Exception $e) {
            $errors['system'] = 'システムエラーが発生しました。もう一度お試しください。';
            error_log("User registration error: " . $e->getMessage());
        }
    }
}

/**
 * ユーザーデータのバリデーション
 */
function validateUserData($data) {
    $errors = [];
    
    // ユーザー名のバリデーション
    if (empty($data['username'])) {
        $errors['username'] = 'ユーザー名は必須です。';
    } elseif (strlen($data['username']) < 3 || strlen($data['username']) > 50) {
        $errors['username'] = 'ユーザー名は3文字以上50文字以下で入力してください。';
    } elseif (!preg_match('/^[a-zA-Z0-9_]+$/', $data['username'])) {
        $errors['username'] = 'ユーザー名は英数字とアンダースコアのみ使用できます。';
    }
    
    // メールアドレスのバリデーション
    if (empty($data['email'])) {
        $errors['email'] = 'メールアドレスは必須です。';
    } elseif (!filter_var($data['email'], FILTER_VALIDATE_EMAIL)) {
        $errors['email'] = '正しいメールアドレス形式で入力してください。';
    } elseif (strlen($data['email']) > 100) {
        $errors['email'] = 'メールアドレスは100文字以下で入力してください。';
    }
    
    // フルネームのバリデーション
    if (empty($data['full_name'])) {
        $errors['full_name'] = 'フルネームは必須です。';
    } elseif (strlen($data['full_name']) > 100) {
        $errors['full_name'] = 'フルネームは100文字以下で入力してください。';
    }
    
    // 生年月日のバリデーション（省略可能）
    if (!empty($data['birth_date'])) {
        $date = DateTime::createFromFormat('Y-m-d', $data['birth_date']);
        if (!$date || $date->format('Y-m-d') !== $data['birth_date']) {
            $errors['birth_date'] = '正しい日付形式（YYYY-MM-DD）で入力してください。';
        } elseif ($date > new DateTime()) {
            $errors['birth_date'] = '未来の日付は選択できません。';
        } elseif ($date < new DateTime('1900-01-01')) {
            $errors['birth_date'] = '1900年以降の日付を選択してください。';
        }
    }
    
    return $errors;
}

/**
 * ユーザー重複チェック
 */
function isUserExists($pdo, $username, $email) {
    $stmt = $pdo->prepare("SELECT COUNT(*) FROM users WHERE username = :username OR email = :email");
    $stmt->execute(['username' => $username, 'email' => $email]);
    return $stmt->fetchColumn() > 0;
}

/**
 * ユーザー挿入処理
 */
function insertUser($pdo, $data) {
    $sql = "INSERT INTO users (username, email, full_name, birth_date) VALUES (:username, :email, :full_name, :birth_date)";
    $stmt = $pdo->prepare($sql);
    return $stmt->execute([
        'username' => $data['username'],
        'email' => $data['email'],
        'full_name' => $data['full_name'],
        'birth_date' => !empty($data['birth_date']) ? $data['birth_date'] : null
    ]);
}

3.2 包括的な入力値検証システム

クライアントサイドとサーバーサイドの両方でのバリデーションを実装し、データの整合性とセキュリティを確保します。

実習 3-2: バリデーション機能の強化

より堅牢なバリデーションシステムとエラーメッセージ表示を実装します。

includes/validation.php作成

バリデーション処理を共通化するためのファイルを作成：

<?php
/**
 * バリデーション共通クラス
 */
class UserValidator {
    
    /**
     * ユーザーデータの包括的なバリデーション
     * @param array $data フォームデータ
     * @return array エラー配列
     */
    public static function validate($data) {
        $errors = [];
        
        // ユーザー名バリデーション
        $username_errors = self::validateUsername($data['username'] ?? '');
        if (!empty($username_errors)) {
            $errors['username'] = $username_errors;
        }
        
        // メールアドレスバリデーション
        $email_errors = self::validateEmail($data['email'] ?? '');
        if (!empty($email_errors)) {
            $errors['email'] = $email_errors;
        }
        
        // フルネームバリデーション
        $fullname_errors = self::validateFullName($data['full_name'] ?? '');
        if (!empty($fullname_errors)) {
            $errors['full_name'] = $fullname_errors;
        }
        
        // 生年月日バリデーション（省略可能）
        if (!empty($data['birth_date'])) {
            $birthdate_errors = self::validateBirthDate($data['birth_date']);
            if (!empty($birthdate_errors)) {
                $errors['birth_date'] = $birthdate_errors;
            }
        }
        
        return $errors;
    }
    
    /**
     * ユーザー名のバリデーション
     */
    private static function validateUsername($username) {
        $username = trim($username);
        
        if (empty($username)) {
            return 'ユーザー名は必須です。';
        }
        
        if (strlen($username) < 3) {
            return 'ユーザー名は3文字以上で入力してください。';
        }
        
        if (strlen($username) > 50) {
            return 'ユーザー名は50文字以下で入力してください。';
        }
        
        if (!preg_match('/^[a-zA-Z0-9_]+$/', $username)) {
            return 'ユーザー名は英数字とアンダースコア(_)のみ使用できます。';
        }
        
        // 予約語チェック
        $reserved_words = ['admin', 'root', 'system', 'user', 'test', 'guest'];
        if (in_array(strtolower($username), $reserved_words)) {
            return 'このユーザー名は予約されているため使用できません。';
        }
        
        return '';
    }
    
    /**
     * メールアドレスのバリデーション
     */
    private static function validateEmail($email) {
        $email = trim($email);
        
        if (empty($email)) {
            return 'メールアドレスは必須です。';
        }
        
        if (strlen($email) > 100) {
            return 'メールアドレスは100文字以下で入力してください。';
        }
        
        if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
            return '正しいメールアドレス形式で入力してください。';
        }
        
        // 危険なドメインのブラックリスト（例）
        $blocked_domains = ['tempmail.com', '10minutemail.com'];
        $domain = substr(strrchr($email, "@"), 1);
        if (in_array(strtolower($domain), $blocked_domains)) {
            return 'このメールドメインは使用できません。';
        }
        
        return '';
    }
    
    /**
     * フルネームのバリデーション
     */
    private static function validateFullName($full_name) {
        $full_name = trim($full_name);
        
        if (empty($full_name)) {
            return 'フルネームは必須です。';
        }
        
        if (strlen($full_name) > 100) {
            return 'フルネームは100文字以下で入力してください。';
        }
        
        // 特殊文字のチェック（基本的な記号のみ許可）
        if (preg_match('/[<>"\']/', $full_name)) {
            return 'フルネームに使用できない文字が含まれています。';
        }
        
        return '';
    }
    
    /**
     * 生年月日のバリデーション
     */
    private static function validateBirthDate($birth_date) {
        $birth_date = trim($birth_date);
        
        // 日付形式チェック
        $date = DateTime::createFromFormat('Y-m-d', $birth_date);
        if (!$date || $date->format('Y-m-d') !== $birth_date) {
            return '正しい日付形式（YYYY-MM-DD）で入力してください。';
        }
        
        // 未来日チェック
        $today = new DateTime();
        if ($date > $today) {
            return '未来の日付は選択できません。';
        }
        
        // 過去日制限チェック
        $min_date = new DateTime('1900-01-01');
        if ($date < $min_date) {
            return '1900年以降の日付を選択してください。';
        }
        
        // 年齢制限チェック（例：150歳以上は無効）
        $age = $today->diff($date)->y;
        if ($age > 150) {
            return '現実的な生年月日を入力してください。';
        }
        
        return '';
    }
    
    /**
     * CSRF対策トークン生成
     */
    public static function generateCsrfToken() {
        if (session_status() == PHP_SESSION_NONE) {
            session_start();
        }
        
        $token = bin2hex(random_bytes(32));
        $_SESSION['csrf_token'] = $token;
        return $token;
    }
    
    /**
     * CSRFトークン検証
     */
    public static function validateCsrfToken($token) {
        if (session_status() == PHP_SESSION_NONE) {
            session_start();
        }
        
        return isset($_SESSION['csrf_token']) && 
               hash_equals($_SESSION['csrf_token'], $token);
    }
}

3.3 安全なデータベース挿入処理

PDOプリペアドステートメントを使用した安全なINSERT処理と、重複チェック機能を実装します。

実習 3-3: ユーザーデータ挿入クラスの作成

データベース操作を担当するUserManagerクラスを作成し、安全性と再利用性を向上させます。

includes/UserManager.php作成
<?php
require_once 'config/database.php';

/**
 * ユーザー管理クラス
 * データベースでのユーザーCRUD操作を担当
 */
class UserManager {
    private $pdo;
    
    public function __construct() {
        $this->pdo = getDatabase();
    }
    
    /**
     * 新規ユーザー登録
     * @param array $user_data ユーザーデータ
     * @return array 結果情報
     */
    public function createUser($user_data) {
        try {
            // トランザクション開始
            $this->pdo->beginTransaction();
            
            // 重複チェック
            if ($this->isUserExists($user_data['username'], $user_data['email'])) {
                $this->pdo->rollBack();
                return [
                    'success' => false,
                    'message' => 'ユーザー名またはメールアドレスが既に使用されています。'
                ];
            }
            
            // ユーザー挿入
            $sql = "INSERT INTO users (username, email, full_name, birth_date, created_at, updated_at) 
                    VALUES (:username, :email, :full_name, :birth_date, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP)";
            
            $stmt = $this->pdo->prepare($sql);
            $result = $stmt->execute([
                'username' => $user_data['username'],
                'email' => $user_data['email'],
                'full_name' => $user_data['full_name'],
                'birth_date' => !empty($user_data['birth_date']) ? $user_data['birth_date'] : null
            ]);
            
            if ($result) {
                $user_id = $this->pdo->lastInsertId();
                $this->pdo->commit();
                
                return [
                    'success' => true,
                    'message' => "ユーザー「{$user_data['full_name']}」を正常に登録しました。",
                    'user_id' => $user_id
                ];
            } else {
                $this->pdo->rollBack();
                return [
                    'success' => false,
                    'message' => 'ユーザー登録に失敗しました。'
                ];
            }
            
        } catch (PDOException $e) {
            $this->pdo->rollBack();
            error_log("User creation error: " . $e->getMessage());
            
            return [
                'success' => false,
                'message' => 'データベースエラーが発生しました。しばらく時間をおいてお試しください。'
            ];
        }
    }
    
    /**
     * ユーザー重複チェック
     * @param string $username
     * @param string $email
     * @return bool
     */
    public function isUserExists($username, $email) {
        $sql = "SELECT COUNT(*) FROM users WHERE username = :username OR email = :email";
        $stmt = $this->pdo->prepare($sql);
        $stmt->execute(['username' => $username, 'email' => $email]);
        return $stmt->fetchColumn() > 0;
    }
    
    /**
     * 個別重複チェック（詳細なエラーメッセージ用）
     * @param string $username
     * @param string $email
     * @return array
     */
    public function checkDuplicates($username, $email) {
        $duplicates = [];
        
        // ユーザー名重複チェック
        $stmt = $this->pdo->prepare("SELECT COUNT(*) FROM users WHERE username = :username");
        $stmt->execute(['username' => $username]);
        if ($stmt->fetchColumn() > 0) {
            $duplicates['username'] = 'このユーザー名は既に使用されています。';
        }
        
        // メールアドレス重複チェック
        $stmt = $this->pdo->prepare("SELECT COUNT(*) FROM users WHERE email = :email");
        $stmt->execute(['email' => $email]);
        if ($stmt->fetchColumn() > 0) {
            $duplicates['email'] = 'このメールアドレスは既に登録されています。';
        }
        
        return $duplicates;
    }
    
    /**
     * 全ユーザー取得
     * @param int $limit
     * @param int $offset
     * @return array
     */
    public function getAllUsers($limit = 50, $offset = 0) {
        $sql = "SELECT id, username, email, full_name, birth_date, created_at 
                FROM users 
                ORDER BY created_at DESC 
                LIMIT :limit OFFSET :offset";
        
        $stmt = $this->pdo->prepare($sql);
        $stmt->bindValue(':limit', $limit, PDO::PARAM_INT);
        $stmt->bindValue(':offset', $offset, PDO::PARAM_INT);
        $stmt->execute();
        
        return $stmt->fetchAll();
    }
    
    /**
     * ユーザー検索
     * @param string $keyword
     * @return array
     */
    public function searchUsers($keyword) {
        $sql = "SELECT id, username, email, full_name, birth_date, created_at 
                FROM users 
                WHERE full_name LIKE :keyword 
                   OR email LIKE :keyword 
                   OR username LIKE :keyword
                ORDER BY created_at DESC";
        
        $stmt = $this->pdo->prepare($sql);
        $stmt->execute(['keyword' => '%' . $keyword . '%']);
        
        return $stmt->fetchAll();
    }
    
    /**
     * ユーザー数カウント
     * @return int
     */
    public function getUserCount() {
        $stmt = $this->pdo->query("SELECT COUNT(*) FROM users");
        return $stmt->fetchColumn();
    }
