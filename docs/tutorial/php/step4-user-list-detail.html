// ページネーション設定
$page = max(1, (int)($_GET['page'] ?? 1));
$per_page = 10;
$offset = ($page - 1) * $per_page;

// 検索キーワード
$search = trim($_GET['search'] ?? '');

try {
    $userManager = new UserManager();
    
    // 検索の有無で処理分岐
    if (!empty($search)) {
        $users = $userManager->searchUsers($search);
        $total_users = count($users);

} catch (Exception $e) {
    $error_message = 'データの取得中にエラーが発生しました。';

複数条件での検索とフィルタリング機能を追加実装します。

UserManagerクラスの拡張

以下のメソッドをUserManager.phpに追加：

/**
 * 高度な検索機能
 * @param array $criteria 検索条件
 * @return array
 */
public function advancedSearch($criteria) {
    $where_conditions = [];
    $params = [];
    
    // キーワード検索
    if (!empty($criteria['keyword'])) {
        $where_conditions[] = "(full_name LIKE :keyword OR email LIKE :keyword OR username LIKE :keyword)";
        $params['keyword'] = '%' . $criteria['keyword'] . '%';
    }
    
    // 年齢範囲検索
    if (!empty($criteria['age_from']) || !empty($criteria['age_to'])) {
        if (!empty($criteria['age_from'])) {
            $birth_year_to = date('Y') - (int)$criteria['age_from'];
            $where_conditions[] = "EXTRACT(YEAR FROM birth_date) <= :birth_year_to";
            $params['birth_year_to'] = $birth_year_to;
        }
        
        if (!empty($criteria['age_to'])) {
            $birth_year_from = date('Y') - (int)$criteria['age_to'];
            $where_conditions[] = "EXTRACT(YEAR FROM birth_date) >= :birth_year_from";
            $params['birth_year_from'] = $birth_year_from;
        }
    }
    
    // 登録日範囲検索
    if (!empty($criteria['date_from'])) {
        $where_conditions[] = "DATE(created_at) >= :date_from";
        $params['date_from'] = $criteria['date_from'];
    }
    
    if (!empty($criteria['date_to'])) {
        $where_conditions[] = "DATE(created_at) <= :date_to";
        $params['date_to'] = $criteria['date_to'];
    }
    
    // SQL構築
    $sql = "SELECT id, username, email, full_name, birth_date, created_at FROM users";
    
    if (!empty($where_conditions)) {
        $sql .= " WHERE " . implode(" AND ", $where_conditions);
    }
    
    $sql .= " ORDER BY created_at DESC";
    
    $stmt = $this->pdo->prepare($sql);
    $stmt->execute($params);
    
    return $stmt->fetchAll();
}

/**
 * ユーザー統計情報の取得
 * @return array
 */
public function getUserStatistics() {
    try {
        // 基本統計
        $total_users = $this->getUserCount();
        
        // 今月の新規登録数
        $stmt = $this->pdo->query("
            SELECT COUNT(*) 
            FROM users 
            WHERE EXTRACT(YEAR FROM created_at) = EXTRACT(YEAR FROM CURRENT_DATE)
              AND EXTRACT(MONTH FROM created_at) = EXTRACT(MONTH FROM CURRENT_DATE)
        ");
        $monthly_new_users = $stmt->fetchColumn();
        
        // 年齢分布
        $stmt = $this->pdo->query("
            SELECT 
                CASE 
                    WHEN birth_date IS NULL THEN '未設定'
                    WHEN EXTRACT(YEAR FROM AGE(birth_date)) < 20 THEN '20歳未満'
                    WHEN EXTRACT(YEAR FROM AGE(birth_date)) < 30 THEN '20-29歳'
                    WHEN EXTRACT(YEAR FROM AGE(birth_date)) < 40 THEN '30-39歳'
                    WHEN EXTRACT(YEAR FROM AGE(birth_date)) < 50 THEN '40-49歳'
                    WHEN EXTRACT(YEAR FROM AGE(birth_date)) < 60 THEN '50-59歳'
                    ELSE '60歳以上'
                END as age_group,
                COUNT(*) as count
            FROM users 
            GROUP BY age_group
            ORDER BY age_group
        ");
        $age_distribution = $stmt->fetchAll();
        
        return [
            'total_users' => $total_users,
            'monthly_new_users' => $monthly_new_users,
            'age_distribution' => $age_distribution
        ];
        
    } catch (PDOException $e) {
        error_log("Statistics error: " . $e->getMessage());
        return [
            'total_users' => 0,
            'monthly_new_users' => 0,
            'age_distribution' => []
        ];
    }
}
高度な検索機能の特徴
  • 複数条件検索: キーワード、年齢、登録日での絞り込み
  • 年齢計算: PostgreSQLのEXTRACT関数を活用
  • 日付範囲検索: 期間指定での検索機能
  • 統計情報: ダッシュボード用のデータ集計
  • 動的SQL構築: 条件に応じたクエリ生成

/**
 * ユーザーIDでの詳細取得
 * @param int $user_id
 * @return array|false
 */
public function getUserById($user_id) {
    $sql = "SELECT id, username, email, full_name, birth_date, created_at, updated_at 
            FROM users 
            WHERE id = :user_id";
    
    $stmt = $this->pdo->prepare($sql);
    $stmt->execute(['user_id' => $user_id]);
    
    return $stmt->fetch();
}
