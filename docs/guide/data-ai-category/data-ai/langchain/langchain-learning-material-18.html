Step 2: document_loader.py - ドキュメント読み込み

src/document_loader.py 
from langchain_community.document_loaders import DirectoryLoader, TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from config import CHUNK_SIZE, CHUNK_OVERLAP
import logging

logger = logging.getLogger(__name__)

def load_documents(data_dir: str):
    """指定ディレクトリからドキュメント読み込み"""
    try:
        loader = DirectoryLoader(
            data_dir,
            glob="**/*.md",  # Markdownファイルのみ
            loader_cls=TextLoader
        )
        documents = loader.load()
        logger.info(f"読み込み完了: {len(documents)}個")
        return documents
    except Exception as e:
        logger.error(f"読み込みエラー: {e}")
        raise

def split_documents(documents):
    """ドキュメントを分割"""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=CHUNK_SIZE,
        chunk_overlap=CHUNK_OVERLAP
    )
    return splitter.split_documents(documents)

Step 3: vectorstore.py - Vector Store管理

src/vectorstore.py 
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import FAISS
from config import VECTORSTORE_PATH, EMBEDDING_MODEL
import os
import logging

logger = logging.getLogger(__name__)

def create_vectorstore(documents):
    """ドキュメントからVector Store作成"""
    embeddings = OpenAIEmbeddings(model=EMBEDDING_MODEL)
    vectorstore = FAISS.from_documents(documents, embeddings)
    logger.info("Vector Store作成完了")
    return vectorstore

def save_vectorstore(vectorstore):
    """Vector Storeをディスクに保存"""
    os.makedirs(os.path.dirname(VECTORSTORE_PATH), exist_ok=True)
    vectorstore.save_local(VECTORSTORE_PATH)
    logger.info(f"保存完了: {VECTORSTORE_PATH}")

def load_vectorstore():
    """保存されたVector Storeを読み込み"""
    embeddings = OpenAIEmbeddings(model=EMBEDDING_MODEL)
    vectorstore = FAISS.load_local(
        VECTORSTORE_PATH,
        embeddings,
        allow_dangerous_deserialization=True
    )
    logger.info("Vector Store読み込み完了")
    return vectorstore

18.4 チャットボット本体の実装

RAG、Memory、LCELを組み合わせて、チャットボットの中核機能を実装します。

chatbot.py - チャットボットクラス

src/chatbot.py 
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain.memory import ConversationBufferMemory
from langchain_core.runnables import RunnablePassthrough
from langchain_core.output_parsers import StrOutputParser
from config import MODEL_NAME, SEARCH_K
import logging

logger = logging.getLogger(__name__)

class TechSupportChatbot:
    """技術サポートチャットボット"""

    def __init__(self, vectorstore):
        self.vectorstore = vectorstore
        self.retriever = vectorstore.as_retriever(
            search_kwargs={"k": SEARCH_K}
        )
        self.memory = ConversationBufferMemory(
            return_messages=True,
            memory_key="chat_history"
        )
        self.chain = self._create_chain()

    def _create_chain(self):
        """LCEL チェーンを構築"""
        llm = ChatOpenAI(model=MODEL_NAME, temperature=0)

        # プロンプトテンプレート
        prompt = ChatPromptTemplate.from_messages([
            ("system", """あなたは技術サポート担当者です。
以下の参考情報を基に、ユーザーの質問に丁寧に回答してください。

参考情報:
{context}

回答の際は以下を心がけてください：
- 参考情報に基づいた正確な回答
- わからない場合は正直に伝える
- コード例は適切にフォーマット
"""),
            MessagesPlaceholder(variable_name="chat_history"),
            ("human", "{question}")
        ])

        # LCEL チェーン
        chain = (
            {
                "context": self.retriever,
                "question": RunnablePassthrough(),
                "chat_history": lambda x: self.memory.chat_memory.messages
            }
            | prompt
            | llm
            | StrOutputParser()
        )
        return chain

    def chat(self, question: str) -> str:
        """ユーザーの質問に回答"""
        try:
            response = self.chain.invoke(question)
            # メモリに保存
            self.memory.chat_memory.add_user_message(question)
            self.memory.chat_memory.add_ai_message(response)
            return response
        except Exception as e:
            logger.error(f"チャットエラー: {e}")
            return f"申し訳ございません。エラーが発生しました: {str(e)}"

    def clear_history(self):
        """会話履歴をクリア"""
        self.memory.clear()
        logger.info("会話履歴をクリアしました")

LCEL チェーンのポイント

contextはRetrieverで検索、questionはそのまま渡し、chat_historyはMemoryから取得します。これらを|で繋ぐLCELの記法により、複雑な処理を簡潔に表現できます。

18.5 Gradio WebUIの実装

Gradioを使って、ユーザーフレンドリーなチャットインターフェースを構築します。

app.py - アプリケーションメイン

app.py 
import gradio as gr
from src.vectorstore import load_vectorstore
from src.chatbot import TechSupportChatbot
import logging

# ログ設定
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/chatbot.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

# チャットボット初期化
vectorstore = load_vectorstore()
chatbot = TechSupportChatbot(vectorstore)

def chat_interface(message, history):
    """Gradio用チャット関数"""
    response = chatbot.chat(message)
    return response

def clear_chat():
    """会話履歴クリア"""
    chatbot.clear_history()
    return None

# Gradio UI構築
with gr.Blocks(theme=gr.themes.Soft()) as demo:
    gr.Markdown("# 🤖 技術サポートチャットボット")
    gr.Markdown("社内ドキュメントを検索して、技術的な質問に回答します。")

    chatbot_ui = gr.ChatInterface(
        fn=chat_interface,
        title="",
        examples=[
            "LangChainとは何ですか？",
            "RAGの実装方法を教えてください",
            "Pythonの仮想環境の作り方は？"
        ],
        retry_btn=None,
        undo_btn=None,
        clear_btn="🗑️ 会話履歴をクリア"
    )

if __name__ == "__main__":
    demo.launch(share=False)

18.6 テストと品質改善

チャットボットの品質を向上させるための、テストとデバッグの手法を学びます。

テスト観点

検索精度テスト

  • • 関連ドキュメントが正しく取得されるか
  • • SEARCH_Kの値は適切か
  • • 類似度スコアの閾値設定

回答品質テスト

  • • 回答が質問に対して適切か
  • • 情報源が明示されているか
  • • 不明時に正直に回答しているか

会話履歴テスト

  • • 文脈を保持した回答ができるか
  • • メモリ容量の管理
  • • 履歴クリア機能の動作

エラーハンドリング

  • • API制限時の挙動
  • • ネットワークエラー時の挙動
  • • 不正な入力への対処

デバッグと改善のヒント

# chatbot.py に追加
def debug_retrieval(self, question: str):
    """検索結果をデバッグ出力"""
    docs = self.retriever.get_relevant_documents(question)
    print(f"\n=== 検索結果 ({len(docs)}件) ===")

        print(f"\n【{i}】")
        print(f"内容: {doc.page_content[:100]}...")
        print(f"ソース: {doc.metadata.get('source', 'N/A')}")

理解度確認クイズ

  1. Q1. このプロジェクトで使用した主要なLangChainコンポーネントを5つ挙げてください。

    回答を表示

    ①Document Loaders（ドキュメント読み込み）、②Text Splitters（テキスト分割）、③Embeddings（ベクトル化）、④Vector Store（FAISS）、⑤Retriever（検索）、⑥ConversationBufferMemory（会話履歴管理）、⑦LCEL（チェーン構築）などです。これらを組み合わせてRAG対応チャットボットを実現しました。

  2. Q2. モジュール分割（document_loader.py、vectorstore.py、chatbot.py）のメリットは何ですか？

    回答を表示

    ①各モジュールを独立してテスト可能、②機能追加・修正が局所化され影響範囲が限定的、③コードの再利用性が向上、④複数人での開発がしやすい、⑤コードの可読性・保守性が向上、などのメリットがあります。大規模なアプリケーションでは必須のプラクティスです。

  3. Q3. 本番環境にデプロイする際に考慮すべきポイントを3つ挙げてください。

    回答を表示

    ①セキュリティ（APIキーの環境変数管理、認証・認可）、②スケーラビリティ（同時接続数対応、Vector Storeの選択）、③モニタリング（エラーログ、パフォーマンス計測、LangSmith活用）、④コスト管理（API呼び出し回数、トークン使用量の監視）、⑤レスポンス速度（キャッシュ、非同期処理）などが重要です。
