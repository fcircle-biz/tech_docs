from sqlmodel import SQLModel, create_engine, Session
from typing import Generator
import os

# 環境変数からデータベースURLを取得
DATABASE_URL = os.getenv("DATABASE_URL", "sqlite:///./app.db")

# エンジンの作成
engine = create_engine(
    DATABASE_URL,
    echo=True,  # 開発時はSQLログを表示
    connect_args={"check_same_thread": False} if "sqlite" in DATABASE_URL else {}
)

def create_db_and_tables():
    """データベースとテーブルを作成"""
    SQLModel.metadata.create_all(engine)

def get_session() -> Generator[Session, None, None]:
    """データベースセッションの依存性注入用ジェネレータ"""
    with Session(engine) as session:
        yield session
実行例 (app/models.py)
from datetime import datetime
from typing import Optional, List
from sqlmodel import SQLModel, Field, Relationship
from enum import Enum

class UserStatus(str, Enum):
    ACTIVE = "active"
    INACTIVE = "inactive"
    SUSPENDED = "suspended"

class User(SQLModel, table=True):
    __tablename__ = "users"
    
    id: Optional[int] = Field(default=None, primary_key=True)
    username: str = Field(min_length=3, max_length=50, unique=True, index=True)
    email: str = Field(max_length=255, unique=True, index=True)
    full_name: Optional[str] = Field(default=None, max_length=200)
    status: UserStatus = Field(default=UserStatus.ACTIVE)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: Optional[datetime] = Field(default=None)
    
    # リレーションシップ
    posts: List["Post"] = Relationship(back_populates="author")

class Post(SQLModel, table=True):
    __tablename__ = "posts"
    
    id: Optional[int] = Field(default=None, primary_key=True)
    title: str = Field(min_length=1, max_length=200)
    content: str = Field(min_length=1)
    is_published: bool = Field(default=False)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: Optional[datetime] = Field(default=None)
    
    # 外部キー
    author_id: int = Field(foreign_key="users.id")
    
    # リレーションシップ
    author: User = Relationship(back_populates="posts")

5.2 RESTful APIエンドポイントの実装

SQLModelとFastAPIを使用して、完全なCRUD操作を提供するRESTful APIを実装します。

実習 5-2: ユーザー管理APIの実装

ユーザーリソースに対する完全なCRUD操作APIを実装します。

実行例 (app/routers/users.py)
from typing import List
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlmodel import Session, select
from ..database import get_session
from ..models import User, UserStatus
from ..schemas import UserRead, UserCreate, UserUpdate

router = APIRouter(prefix="/users", tags=["users"])

@router.post("/", response_model=UserRead, status_code=201)
def create_user(
    user_data: UserCreate,
    session: Session = Depends(get_session)
):
    """新しいユーザーを作成"""
    # 重複チェック
    existing_user = session.exec(
        select(User).where(
            (User.username == user_data.username) | 
            (User.email == user_data.email)
        )
    ).first()
    
    if existing_user:
        raise HTTPException(
            status_code=400,
            detail="ユーザー名またはメールアドレスが既に使用されています"
        )
    
    # ユーザー作成
    user = User(**user_data.dict())
    session.add(user)
    session.commit()
    session.refresh(user)
    
    return user

@router.get("/", response_model=List[UserRead])
def get_users(
    skip: int = Query(0, ge=0, description="スキップする件数"),
    limit: int = Query(100, ge=1, le=1000, description="取得する件数"),
    status: Optional[UserStatus] = Query(None, description="ステータスでフィルタ"),
    session: Session = Depends(get_session)
):
    """ユーザー一覧を取得"""
    statement = select(User)
    
    # ステータスでフィルタ
    if status:
        statement = statement.where(User.status == status)
    
    # ページネーション
    statement = statement.offset(skip).limit(limit)
    
    users = session.exec(statement).all()
    return users

@router.get("/{user_id}", response_model=UserRead)
def get_user(
    user_id: int,
    session: Session = Depends(get_session)
):
    """指定されたIDのユーザーを取得"""
    user = session.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="ユーザーが見つかりません")
    return user

@router.put("/{user_id}", response_model=UserRead)
def update_user(
    user_id: int,
    user_data: UserUpdate,
    session: Session = Depends(get_session)
):
    """ユーザー情報を更新"""
    user = session.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="ユーザーが見つかりません")
    
    # 更新データを適用
    update_data = user_data.dict(exclude_unset=True)
    for field, value in update_data.items():
        setattr(user, field, value)
    
    # 更新時刻を設定
    user.updated_at = datetime.utcnow()
    
    session.add(user)
    session.commit()
    session.refresh(user)
    
    return user

@router.delete("/{user_id}", status_code=204)
def delete_user(
    user_id: int,
    session: Session = Depends(get_session)
):
    """ユーザーを削除"""
    user = session.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="ユーザーが見つかりません")
    
    session.delete(user)
    session.commit()
    
    return None

@router.get("/{user_id}/posts", response_model=List[PostRead])
def get_user_posts(
    user_id: int,
    published_only: bool = Query(True, description="公開済みの投稿のみ"),
    session: Session = Depends(get_session)
):
    """ユーザーの投稿一覧を取得"""
    user = session.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="ユーザーが見つかりません")
    
    statement = select(Post).where(Post.author_id == user_id)
    
    if published_only:
        statement = statement.where(Post.is_published == True)
    
    posts = session.exec(statement).all()
    return posts

5.4 自動バリデーションとエラーハンドリング

SQLModelとFastAPIの組み合わせにより、自動的なバリデーションとエラーハンドリングが実現されます。

実習 5-4: 高度なバリデーションとエラーハンドリング

カスタムバリデーションとエラーハンドリングを実装します。

実行例 (app/exceptions.py)
from fastapi import HTTPException, Request
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError
from pydantic import ValidationError
from sqlalchemy.exc import IntegrityError
import logging

logger = logging.getLogger(__name__)

class APIException(HTTPException):
    """カスタムAPI例外"""
    def __init__(self, status_code: int, detail: str, error_code: str = None):
        super().__init__(status_code=status_code, detail=detail)
        self.error_code = error_code

class UserNotFoundError(APIException):
    def __init__(self, user_id: int):
        super().__init__(
            status_code=404,
            detail=f"ユーザーID {user_id} が見つかりません",
            error_code="USER_NOT_FOUND"
        )

class DuplicateUserError(APIException):
    def __init__(self, field: str, value: str):
        super().__init__(
            status_code=409,
            detail=f"{field}: '{value}' は既に使用されています",
            error_code="DUPLICATE_USER"
        )

# グローバル例外ハンドラ
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    """バリデーションエラーのハンドリング"""
    logger.error(f"Validation error: {exc.errors()}")
    
    return JSONResponse(
        status_code=422,
        content={
            "error": "バリデーションエラー",
            "details": exc.errors(),
            "error_code": "VALIDATION_ERROR"
        }
    )

async def integrity_error_handler(request: Request, exc: IntegrityError):
    """データベース整合性エラーのハンドリング"""
    logger.error(f"Database integrity error: {str(exc)}")
    
    return JSONResponse(
        status_code=409,
        content={
            "error": "データの整合性エラーが発生しました",
            "error_code": "INTEGRITY_ERROR"
        }
    )

async def api_exception_handler(request: Request, exc: APIException):
    """カスタムAPI例外のハンドリング"""
    return JSONResponse(
        status_code=exc.status_code,
        content={
            "error": exc.detail,
            "error_code": exc.error_code
        }
    )
メインアプリケーションでの例外ハンドラ登録
# app/main.py に追加
from .exceptions import (
    validation_exception_handler,
    integrity_error_handler,
    api_exception_handler,
    APIException
)

# 例外ハンドラの登録
app.add_exception_handler(RequestValidationError, validation_exception_handler)
app.add_exception_handler(IntegrityError, integrity_error_handler)
app.add_exception_handler(APIException, api_exception_handler)

実習 5-5: APIテストの作成

pytest と FastAPI のテストクライアントを使用してAPIテストを実装します。

実行例 (tests/test_users.py)
import pytest
from fastapi.testclient import TestClient
from sqlmodel import SQLModel, create_engine, Session
from sqlmodel.pool import StaticPool
from app.main import app
from app.database import get_session
from app.models import User

# テスト用のインメモリデータベース
@pytest.fixture(name="session")
def session_fixture():
    engine = create_engine(
        "sqlite:///:memory:",
        connect_args={"check_same_thread": False},
        poolclass=StaticPool,
    )
    SQLModel.metadata.create_all(engine)
    with Session(engine) as session:
        yield session

@pytest.fixture(name="client")
def client_fixture(session: Session):
    def get_session_override():
        return session
    
    app.dependency_overrides[get_session] = get_session_override
    client = TestClient(app)
    yield client
    app.dependency_overrides.clear()

def test_create_user(client: TestClient):
    """ユーザー作成のテスト"""
    response = client.post(
        "/api/v1/users/",
        json={
            "username": "testuser",
            "email": "test@example.com",
            "full_name": "Test User"
        }
    )
    assert response.status_code == 201
    data = response.json()
    assert data["username"] == "testuser"
    assert data["email"] == "test@example.com"
    assert "id" in data

def test_get_users(client: TestClient, session: Session):
    """ユーザー一覧取得のテスト"""
    # テストユーザーを作成
    user = User(username="testuser", email="test@example.com")
    session.add(user)
    session.commit()
    
    response = client.get("/api/v1/users/")
    assert response.status_code == 200
    data = response.json()
    assert len(data) == 1
    assert data[0]["username"] == "testuser"

def test_get_user_not_found(client: TestClient):
    """存在しないユーザーの取得テスト"""
    response = client.get("/api/v1/users/999")
    assert response.status_code == 404

def test_duplicate_user_creation(client: TestClient):
    """重複ユーザー作成のテスト"""
    # 最初のユーザーを作成
    client.post(
        "/api/v1/users/",
        json={"username": "duplicate", "email": "duplicate@example.com"}
    )
    
    # 同じユーザー名で再度作成を試行
    response = client.post(
        "/api/v1/users/",
        json={"username": "duplicate", "email": "different@example.com"}
    )
    assert response.status_code == 400
