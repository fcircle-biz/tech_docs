6.1 Route Handlersの基本

6.1.1 Route Handlersとは

Route Handlersは、App RouterにおけるAPIエンドポイントの作成方法です。app/apiディレクトリ内にroute.jsファイルを配置することで、APIを作成できます。

flowchart LR A["クライアント"] --> B["/api/users"] B --> C["app/api/users/route.js"] C --> D["JSONレスポンス"] D --> A

6.1.2 GETリクエストの実装

// app/api/hello/route.js
export async function GET(request) {
  return Response.json({
    message: 'Hello from Next.js API!',
    timestamp: new Date().toISOString()
  });
}

// アクセス: http://localhost:3000/api/hello

6.2 HTTPメソッドの実装

6.2.1 各HTTPメソッドの用途

メソッド 用途 例
GETデータの取得ユーザー一覧の取得
POST新規データの作成新しいユーザーの登録
PUT/PATCHデータの更新ユーザー情報の変更
DELETEデータの削除ユーザーの削除
実習 6-1: 完全なCRUD APIの作成
// app/api/todos/route.js
let todos = [
  { id: 1, title: 'Next.jsを学ぶ', completed: false },
  { id: 2, title: 'APIを作成する', completed: false },
];

// GET: すべてのTodoを取得
export async function GET(request) {
  return Response.json(todos);
}

// POST: 新しいTodoを作成
export async function POST(request) {
  const body = await request.json();
  const newTodo = {
    id: todos.length + 1,
    title: body.title,
    completed: false
  };
  todos.push(newTodo);
  return Response.json(newTodo, { status: 201 });
}
// app/api/todos/[id]/route.js
// GET: 特定のTodoを取得
export async function GET(request, { params }) {
  const todo = todos.find(t => t.id === parseInt(params.id));
  if (!todo) {
    return Response.json({ error: 'Not found' }, { status: 404 });
  }
  return Response.json(todo);
}

// PUT: Todoを更新
export async function PUT(request, { params }) {
  const body = await request.json();
  const index = todos.findIndex(t => t.id === parseInt(params.id));
  if (index === -1) {
    return Response.json({ error: 'Not found' }, { status: 404 });
  }
  todos[index] = { ...todos[index], ...body };
  return Response.json(todos[index]);
}

// DELETE: Todoを削除
export async function DELETE(request, { params }) {
  const index = todos.findIndex(t => t.id === parseInt(params.id));
  if (index === -1) {
    return Response.json({ error: 'Not found' }, { status: 404 });
  }
  todos.splice(index, 1);
  return Response.json({ message: 'Deleted successfully' });
}

6.3 Server Actions

6.3.1 Server Actionsとは

Server Actionsは、サーバー側で実行される関数で、フォームからの直接呼び出しやクライアントコンポーネントからの実行が可能です。従来のAPIエンドポイントを作成せずにサーバー側の処理を実行できます。

6.3.2 基本的なServer Action

// app/actions.js
'use server';

export async function createUser(formData) {
  const name = formData.get('name');
  const email = formData.get('email');

  // データベースへの保存（例）
  // await db.users.create({ name, email });

  console.log('新しいユーザー:', name, email);
  return { success: true, message: 'ユーザーを作成しました' };
}

6.3.3 フォームでのServer Action使用

// app/register/page.js
import { createUser } from '../actions';

export default function RegisterPage() {
  return (
    <form action={createUser}>
      <input type="text" name="name" placeholder="名前" required />
      <input type="email" name="email" placeholder="メール" required />
      <button type="submit">登録</button>
    </form>
  );
}

6.3.4 クライアントコンポーネントからの呼び出し

// app/components/UserForm.js
'use client';

import { createUser } from '../actions';
import { useState } from 'react';

export default function UserForm() {
  const [message, setMessage] = useState('');

  const handleSubmit = async (e) => {
    e.preventDefault();
    const formData = new FormData(e.target);
    const result = await createUser(formData);
    setMessage(result.message);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input type="text" name="name" placeholder="名前" />
      <input type="email" name="email" placeholder="メール" />
      <button type="submit">登録</button>
      {message && <p>{message}</p>}
    </form>
  );
}

6.4 エラーハンドリングとバリデーション

6.4.1 入力バリデーション

// app/actions.js
'use server';

export async function createPost(formData) {
  const title = formData.get('title');
  const content = formData.get('content');

  // バリデーション
  if (!title || title.length < 3) {
    return { 
      success: false, 
      error: 'タイトルは3文字以上必要です' 
    };
  }

  if (!content || content.length < 10) {
    return { 
      success: false, 
      error: 'コンテンツは10文字以上必要です' 
    };
  }

  // データベースへの保存（例）
  try {
    // await db.posts.create({ title, content });
    return { 
      success: true, 
      message: '投稿を作成しました' 
    };
  } catch (error) {
    return { 
      success: false, 
      error: 'データベースエラーが発生しました' 
    };
  }
}

6.4.2 revalidatePathでキャッシュ更新

Server Actionでデータを変更した後、ページのキャッシュを更新するにはrevalidatePathを使用します：

// app/actions.js
'use server';

import { revalidatePath } from 'next/cache';

export async function createPost(formData) {
  const title = formData.get('title');
  const content = formData.get('content');

  // データベースへの保存
  // await db.posts.create({ title, content });

  // /blogページのキャッシュを再検証
  revalidatePath('/blog');

  return { success: true };
}
理解度確認クイズ
  1. App Routerでのアルルアンドポイントを作成するファイル名は何ですか？
    回答：route.js（app/apiディレクトリ内に配置）
  2. Server Actionを定義するファイルの先頭に必要なディレクティブは何ですか？
    回答：'use server'
  3. データ変更後にページキャッシュを更新する関数は何ですか？
    回答：revalidatePath()
