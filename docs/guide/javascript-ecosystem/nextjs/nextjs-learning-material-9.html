9.1 コード分割とダイナミックインポート

9.1.1 ダイナミックインポートの基本

next/dynamicを使うと、コンポーネントを必要な時だけ読み込めます： 

// app/page.js
import dynamic from 'next/dynamic';

// Chart コンポーネントを動的にインポート
const Chart = dynamic(() => import('./components/Chart'), {
  loading: () => <p>チャートを読み込み中...</p>,
  ssr: false  // クライアント側でのみ読み込む
});

export default function DashboardPage() {
  return (
    <div>
      <h1>ダッシュボード</h1>
      <Chart data={[1, 2, 3, 4]} />
    </div>
  );
}

9.2 メモ化によるパフォーマンス改善

9.2.1 React.memo

コンポーネントの不要な再レンダリングを防ぎます：

// app/components/ExpensiveComponent.js
'use client';

import { memo } from 'react';

const ExpensiveComponent = memo(function ExpensiveComponent({ data }) {
  // 重い処理
  console.log('ExpensiveComponent rendered');
  return <div>{data}</div>;
});

export default ExpensiveComponent;

9.2.2 useMemoとuseCallback

'use client';

import { useState, useMemo, useCallback } from 'react';

export default function OptimizedComponent() {
  const [count, setCount] = useState(0);
  const [items, setItems] = useState([1, 2, 3, 4, 5]);

  // 重い計算をメモ化
  const expensiveValue = useMemo(() => {
    console.log('計算実行');
    return items.reduce((a, b) => a + b, 0);
  }, [items]);

  // 関数をメモ化
  const handleClick = useCallback(() => {
    setCount(c => c + 1);
  }, []);

  return (
    <div>
      <p>合計: {expensiveValue}</p>
      <p>カウント: {count}</p>
      <button onClick={handleClick}>+1</button>
    </div>
  );
}

9.3 Core Web Vitals

指標 意味 目標値
LCP（Largest Contentful Paint） 最大コンテンツの表示時間 2.5秒以下
FID（First Input Delay） 最初の入力遅延 100ms以下
CLS（Cumulative Layout Shift） レイアウトのずれ 0.1以下

9.4 エラーハンドリング

9.4.1 error.jsによるエラーバウンダリ

// app/blog/error.js
'use client';

export default function Error({ error, reset }) {
  return (
    <div>
      <h2>エラーが発生しました</h2>
      <p>{error.message}</p>
      <button onClick={() => reset()}>再試行</button>
    </div>
  );
}

9.5 Metadata APIによるSEO最適化

// app/blog/[slug]/page.js
export async function generateMetadata({ params }) {
  const post = await fetchPost(params.slug);

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      images: [post.image],
    },
  };
}

export default async function BlogPost({ params }) {
  const post = await fetchPost(params.slug);
  return <article>{post.content}</article>;
}
