6.1 オブジェクト指向プログラミングとは

オブジェクト指向プログラミング（OOP）は、データ（プロパティ）と処理（メソッド）を一つのまとまり（オブジェクト）として扱うプログラミング手法です。

オブジェクト指向の利点

  • 再利用性: 一度作成したクラスを何度でも使用できる
  • 保守性: 機能が整理され、修正が容易
  • 拡張性: 既存のクラスを継承して新しい機能を追加
  • カプセル化: 内部の実装を隠蔽し、安全な操作を提供

現実世界の例

「車」を例にすると：

  • クラス: 車の設計図（共通の仕様）
  • オブジェクト: 実際の車（具体的な個体）
  • プロパティ: 色、速度、燃料残量
  • メソッド: 発進、停止、方向転換

6.2 クラスとオブジェクトの基本

シンプルなクラスの定義

<?php
// クラスの定義
class Car {
    // プロパティ（属性）
    public $color;
    public $speed;
    public $fuel;
    
    // メソッド（機能）
    public function start() {
        return "車が発進しました";
    }
    
    public function stop() {
        return "車が停止しました";
    }
    
    public function getInfo() {
        return "色: {$this->color}, 速度: {$this->speed}km/h, 燃料: {$this->fuel}L";
    }
}

// オブジェクトの作成
$myCar = new Car();

// プロパティの設定
$myCar->color = "赤";
$myCar->speed = 0;
$myCar->fuel = 50;

// メソッドの呼び出し
echo $myCar->start();
echo $myCar->getInfo();

6.3 コンストラクタとデストラクタ

コンストラクタ

コンストラクタは、オブジェクトが作成されるときに自動的に呼び出される特別なメソッドです。

<?php
class Person {
    public $name;
    public $age;
    
    // コンストラクタ
    public function __construct($name, $age) {
        $this->name = $name;
        $this->age = $age;
        echo "新しいPersonオブジェクトが作成されました: {$name}<br>";
    }
    
    public function introduce() {
        return "私の名前は{$this->name}で、{$this->age}歳です。";
    }
}

// オブジェクトの作成（コンストラクタが自動実行）
$person1 = new Person("田中太郎", 25);
$person2 = new Person("佐藤花子", 30);

echo $person1->introduce();
echo $person2->introduce();
実習 6-1: 銀行口座クラス

銀行口座を表すクラスを作成し、残高の管理を行いましょう。

実行例
<?php
class BankAccount {
    private $accountNumber;
    private $balance;
    private $ownerName;
    
    public function __construct($accountNumber, $ownerName, $initialBalance = 0) {
        $this->accountNumber = $accountNumber;
        $this->ownerName = $ownerName;
        $this->balance = $initialBalance;
    }
    
    // 入金
    public function deposit($amount) {
        if ($amount > 0) {
            $this->balance += $amount;
            return "¥" . number_format($amount) . "を入金しました。残高: ¥" . number_format($this->balance);
        }
        return "入金額は正の値である必要があります。";
    }
    
    // 出金
    public function withdraw($amount) {
        if ($amount > 0 && $amount <= $this->balance) {
            $this->balance -= $amount;
            return "¥" . number_format($amount) . "を出金しました。残高: ¥" . number_format($this->balance);
        }
        return "出金できません。残高不足または無効な金額です。";
    }
    
    // 残高照会
    public function getBalance() {
        return "現在の残高: ¥" . number_format($this->balance);
    }
    
    // 口座情報
    public function getAccountInfo() {
        return "口座番号: {$this->accountNumber}, 名義: {$this->ownerName}, 残高: ¥" . number_format($this->balance);
    }
}

// 銀行口座の作成
$account1 = new BankAccount("123-456789", "田中太郎", 100000);
$account2 = new BankAccount("987-654321", "佐藤花子", 50000);

echo "<h3>口座情報</h3>";
echo "<p>" . $account1->getAccountInfo() . "</p>";
echo "<p>" . $account2->getAccountInfo() . "</p>";

echo "<h3>取引履歴</h3>";
echo "<p>" . $account1->deposit(25000) . "</p>";
echo "<p>" . $account1->withdraw(30000) . "</p>";
echo "<p>" . $account2->withdraw(100000) . "</p>";  // 残高不足
echo "<p>" . $account2->deposit(20000) . "</p>";
期待される結果

口座情報の表示と、入金・出金の取引が正常に処理され、残高不足時にはエラーメッセージが表示されます。

6.4 アクセス修飾子

アクセス修飾子は、プロパティやメソッドへのアクセスを制御します。

修飾子 アクセス範囲 説明
public どこからでも クラスの外部からアクセス可能
private 同じクラス内のみ クラスの内部からのみアクセス可能
protected 同じクラスと継承先 継承関係にあるクラスからアクセス可能 
<?php
class Student {
    public $name;           // 外部からアクセス可能
    private $id;            // クラス内部からのみアクセス可能
    protected $grade;       // 継承先からもアクセス可能
    
    public function __construct($name, $id, $grade) {
        $this->name = $name;
        $this->id = $id;
        $this->grade = $grade;
    }
    
    // privateプロパティにアクセスするためのメソッド
    public function getId() {
        return $this->id;
    }
    
    public function setGrade($grade) {
        $this->grade = $grade;
    }
}

$student = new Student("田中", "S001", "A");
echo $student->name;        // OK: publicプロパティ
echo $student->getId();     // OK: publicメソッド経由
// echo $student->id;       // エラー: privateプロパティに直接アクセス不可

6.5 継承

継承により、既存のクラスの機能を受け継いで新しいクラスを作成できます。

<?php
// 基底クラス（親クラス）
class Animal {
    protected $name;
    protected $age;
    
    public function __construct($name, $age) {
        $this->name = $name;
        $this->age = $age;
    }
    
    public function eat() {
        return "{$this->name}が食事をしています。";
    }
    
    public function sleep() {
        return "{$this->name}が眠っています。";
    }
}

// 派生クラス（子クラス）
class Dog extends Animal {
    private $breed;
    
    public function __construct($name, $age, $breed) {
        parent::__construct($name, $age);  // 親のコンストラクタを呼び出し
        $this->breed = $breed;
    }
    
    public function bark() {
        return "{$this->name}が吠えています：ワンワン！";
    }
    
    public function getInfo() {
        return "犬の名前: {$this->name}, 年齢: {$this->age}歳, 種類: {$this->breed}";
    }
}

// 使用例
$dog = new Dog("ポチ", 3, "柴犬");
echo $dog->eat();      // 親クラスのメソッド
echo $dog->bark();     // 子クラス独自のメソッド
echo $dog->getInfo();
実習 6-2: 図形クラスの継承

図形を表す基底クラスを作成し、円と四角形のクラスを継承で実装しましょう。

実行例
<?php
// 基底クラス
abstract class Shape {
    protected $name;
    
    public function __construct($name) {
        $this->name = $name;
    }
    
    public function getName() {
        return $this->name;
    }
    
    // 抽象メソッド（継承先で必ず実装）
    abstract public function getArea();
    abstract public function getPerimeter();
}

// 円クラス
class Circle extends Shape {
    private $radius;
    
    public function __construct($radius) {
        parent::__construct("円");
        $this->radius = $radius;
    }
    
    public function getArea() {
        return pi() * $this->radius * $this->radius;
    }
    
    public function getPerimeter() {
        return 2 * pi() * $this->radius;
    }
}

// 四角形クラス
class Rectangle extends Shape {
    private $width;
    private $height;
    
    public function __construct($width, $height) {
        parent::__construct("四角形");
        $this->width = $width;
        $this->height = $height;
    }
    
    public function getArea() {
        return $this->width * $this->height;
    }
    
    public function getPerimeter() {
        return 2 * ($this->width + $this->height);
    }
}

// 図形の配列
$shapes = [
    new Circle(5),
    new Rectangle(4, 6),
    new Circle(3),
    new Rectangle(8, 3)
];

echo "<h3>図形の情報</h3>";
foreach ($shapes as $shape) {
    echo "<p>";
    echo "{$shape->getName()}: ";
    echo "面積 = " . round($shape->getArea(), 2) . ", ";
    echo "周囲 = " . round($shape->getPerimeter(), 2);
    echo "</p>";
}
期待される結果

各図形の名前、面積、周囲の長さが計算されて表示されます。継承により共通の処理と個別の処理が適切に分離されています。

6.6 静的メソッドとプロパティ

静的メソッドとプロパティは、インスタンスを作成せずにクラスから直接アクセスできます。

<?php
class MathHelper {
    // 静的プロパティ
    public static $pi = 3.14159;
    
    // 静的メソッド
    public static function add($a, $b) {
        return $a + $b;
    }
    
    public static function multiply($a, $b) {
        return $a * $b;
    }
    
    public static function circleArea($radius) {
        return self::$pi * $radius * $radius;
    }
}

// 静的メソッドの使用（インスタンス作成不要）
echo MathHelper::add(5, 3);                 // 8
echo MathHelper::multiply(4, 7);             // 28
echo MathHelper::circleArea(5);              // 78.54
echo MathHelper::$pi;                        // 3.14159
