7.1 Alembicの導入とセットアップ

AlembicはSQLAlchemyのデータベースマイグレーションツールです。SQLModelでもAlembicを使用してスキーマ変更を安全に管理できます。

# alembic/env.py
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context
import os
import sys

# プロジェクトルートをパスに追加
sys.path.append(os.path.dirname(os.path.dirname(os.path.realpath(__file__))))

# SQLModelのインポート
from app.models import SQLModel

# Alembic設定オブジェクト
config = context.config

# ログ設定
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

# MetaDataオブジェクトの設定
target_metadata = SQLModel.metadata

def get_url():
    """データベースURLを環境変数から取得"""
    url = os.getenv("DATABASE_URL", "sqlite:///./app.db")
    return url

def run_migrations_offline() -> None:
    """オフラインモードでのマイグレーション実行"""
    url = get_url()
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )

    with context.begin_transaction():
        context.run_migrations()

def run_migrations_online() -> None:
    """オンラインモードでのマイグレーション実行"""
    configuration = config.get_section(config.config_ini_section)
    configuration["sqlalchemy.url"] = get_url()
    
    connectable = engine_from_config(
        configuration,
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )

    with connectable.connect() as connection:
        context.configure(
            connection=connection, 
            target_metadata=target_metadata
        )

        with context.begin_transaction():
            context.run_migrations()

if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()

7.4 本番環境でのマイグレーション戦略

本番環境でのマイグレーション実行は慎重に行う必要があります。ダウンタイムを最小限に抑え、安全にスキーマ変更を適用する戦略を学習します。

実習 7-4: 本番環境マイグレーション戦略

本番環境での安全なマイグレーション戦略を実装します。

実行例 (deployment/migrate.py)
#!/usr/bin/env python3
"""
本番環境用マイグレーションスクリプト
"""

import os
import sys
import subprocess
import logging
from datetime import datetime
from pathlib import Path

# ログ設定
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler(f'migration_{datetime.now().strftime("%Y%m%d_%H%M%S")}.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

class ProductionMigrator:
    """本番環境用マイグレーター"""
    
    def __init__(self, db_url: str, backup_path: str = "backups/"):
        self.db_url = db_url
        self.backup_path = Path(backup_path)
        self.backup_path.mkdir(exist_ok=True)
    
    def create_backup(self) -> str:
        """データベースバックアップを作成"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_file = self.backup_path / f"backup_{timestamp}.sql"
        
        logger.info(f"データベースバックアップを作成中: {backup_file}")
        
        try:
            # PostgreSQLの場合の例
            if "postgresql" in self.db_url:
                cmd = f"pg_dump {self.db_url} > {backup_file}"
            # MySQLの場合の例
            elif "mysql" in self.db_url:
                cmd = f"mysqldump --single-transaction {self.db_url} > {backup_file}"
            # SQLiteの場合の例
            else:
                # SQLiteデータベースファイルをコピー
                import shutil
                db_file = self.db_url.replace('sqlite:///', '')
                shutil.copy2(db_file, backup_file.with_suffix('.db'))
                return str(backup_file.with_suffix('.db'))
            
            result = subprocess.run(cmd, shell=True, check=True, capture_output=True, text=True)
            logger.info("バックアップが正常に作成されました")
            return str(backup_file)
            
        except subprocess.CalledProcessError as e:
            logger.error(f"バックアップ作成に失敗: {e}")
            raise
    
    def dry_run_migration(self) -> bool:
        """マイグレーションのドライラン"""
        logger.info("マイグレーションのドライランを実行中...")
        
        try:
            # 現在のマイグレーション状態を確認
            result = subprocess.run(
                ["alembic", "current"], 
                capture_output=True, 
                text=True, 
                check=True
            )
            current_revision = result.stdout.strip()
            logger.info(f"現在のリビジョン: {current_revision}")
            
            # 利用可能なマイグレーションを確認
            result = subprocess.run(
                ["alembic", "history", "--verbose"], 
                capture_output=True, 
                text=True, 
                check=True
            )
            logger.info(f"マイグレーション履歴:\n{result.stdout}")
            
            # マイグレーションSQLを生成（実行はしない）
            result = subprocess.run(
                ["alembic", "upgrade", "head", "--sql"], 
                capture_output=True, 
                text=True, 
                check=True
            )
            
            if result.stdout.strip():
                logger.info(f"実行予定のSQL:\n{result.stdout}")
                return True
            else:
                logger.info("適用すべきマイグレーションはありません")
                return False
                
        except subprocess.CalledProcessError as e:
            logger.error(f"ドライラン実行エラー: {e}")
            return False
    
    def execute_migration(self) -> bool:
        """実際のマイグレーション実行"""
        logger.info("マイグレーションを実行中...")
        
        try:
            result = subprocess.run(
                ["alembic", "upgrade", "head"], 
                capture_output=True, 
                text=True, 
                check=True
            )
            
            logger.info("マイグレーションが正常に完了しました")
            logger.info(f"出力: {result.stdout}")
            return True
            
        except subprocess.CalledProcessError as e:
            logger.error(f"マイグレーション実行エラー: {e}")
            logger.error(f"エラー出力: {e.stderr}")
            return False
    
    def rollback_migration(self, target_revision: str) -> bool:
        """マイグレーションのロールバック"""
        logger.warning(f"リビジョン {target_revision} にロールバック中...")
        
        try:
            result = subprocess.run(
                ["alembic", "downgrade", target_revision], 
                capture_output=True, 
                text=True, 
                check=True
            )
            
            logger.info("ロールバックが正常に完了しました")
            return True
            
        except subprocess.CalledProcessError as e:
            logger.error(f"ロールバック実行エラー: {e}")
            return False

def main():
    """メイン実行関数"""
    if len(sys.argv) < 2:
        print("使用法: python migrate.py [backup|dry-run|migrate|rollback ]")
        sys.exit(1)
    
    # 環境変数からデータベースURLを取得
    db_url = os.getenv("DATABASE_URL")
    if not db_url:
        logger.error("DATABASE_URL環境変数が設定されていません")
        sys.exit(1)
    
    migrator = ProductionMigrator(db_url)
    command = sys.argv[1]
    
    if command == "backup":
        backup_file = migrator.create_backup()
        print(f"バックアップが作成されました: {backup_file}")
        
    elif command == "dry-run":
        has_migrations = migrator.dry_run_migration()
        if not has_migrations:
            print("適用すべきマイグレーションはありません")
        
    elif command == "migrate":
        # 1. バックアップ作成
        backup_file = migrator.create_backup()
        
        # 2. ドライラン
        if not migrator.dry_run_migration():
            print("適用すべきマイグレーションはありません")
            return
        
        # 3. 確認
        response = input("マイグレーションを実行しますか？ (yes/no): ")
        if response.lower() != 'yes':
            print("マイグレーションをキャンセルしました")
            return
        
        # 4. 実行
        if migrator.execute_migration():
            print("マイグレーションが正常に完了しました")
        else:
            print("マイグレーションに失敗しました。バックアップから復旧してください")
            print(f"バックアップファイル: {backup_file}")
            
    elif command == "rollback":
        if len(sys.argv) < 3:
            print("使用法: python migrate.py rollback ")
            sys.exit(1)
        
        revision = sys.argv[2]
        response = input(f"リビジョン {revision} にロールバックしますか？ (yes/no): ")
        if response.lower() != 'yes':
            print("ロールバックをキャンセルしました")
            return
        
        if migrator.rollback_migration(revision):
            print("ロールバックが正常に完了しました")
        else:
            print("ロールバックに失敗しました")
            
    else:
        print(f"不明なコマンド: {command}")
        sys.exit(1)

if __name__ == "__main__":
    main()
本番環境でのマイグレーション実行例

7.5 データベース監視と最適化

本番環境でのデータベースパフォーマンス監視と最適化手法を学習します。

実習 7-5: データベース監視システムの構築

データベースの健康状態を監視し、パフォーマンス問題を早期発見するシステムを構築します。

実行例 (monitoring/db_monitor.py)
#!/usr/bin/env python3
"""
データベース監視システム
"""

import time
import logging
import psutil
from datetime import datetime, timedelta
from typing import Dict, Any, List
from sqlmodel import Session, create_engine, text
from dataclasses import dataclass

@dataclass
class DatabaseMetrics:
    """データベースメトリクス"""
    timestamp: datetime
    active_connections: int
    slow_queries: int
    table_sizes: Dict[str, int]
    index_usage: Dict[str, float]
    cache_hit_ratio: float
    disk_usage: float
    cpu_usage: float
    memory_usage: float

class DatabaseMonitor:
    """データベース監視クラス"""
    
    def __init__(self, db_url: str):
        self.engine = create_engine(db_url, echo=False)
        self.logger = logging.getLogger(__name__)
    
    def get_connection_count(self, session: Session) -> int:
        """アクティブな接続数を取得"""
        try:
            # PostgreSQLの例
            result = session.exec(text("""
                SELECT count(*) FROM pg_stat_activity 
                WHERE state = 'active'
            """)).one()
            return result
        except Exception:
            # SQLiteやその他のDBの場合は常に1を返す
            return 1
    
    def get_slow_queries(self, session: Session) -> int:
        """遅いクエリの数を取得"""
        try:
            # PostgreSQLの例（仮想的）
            result = session.exec(text("""
                SELECT count(*) FROM pg_stat_statements 
                WHERE mean_time > 1000
            """)).one()
            return result
        except Exception:
            return 0
    
    def get_table_sizes(self, session: Session) -> Dict[str, int]:
        """テーブルサイズを取得"""
        sizes = {}
        try:
            # SQLiteの例
            tables = ['users', 'posts', 'tags', 'post_tag_links']
            for table in tables:
                result = session.exec(text(f"SELECT count(*) FROM {table}")).one()
                sizes[table] = result
        except Exception as e:
            self.logger.warning(f"テーブルサイズ取得エラー: {e}")
        
        return sizes
    
    def get_index_usage(self, session: Session) -> Dict[str, float]:
        """インデックス使用率を取得"""
        usage = {}
        try:
            # PostgreSQLの例（簡略化）
            result = session.exec(text("""
                SELECT 
                    schemaname,
                    tablename,
                    indexname,
                    idx_scan,
                    idx_tup_read,
                    idx_tup_fetch
                FROM pg_stat_user_indexes
            """)).all()
            
            for row in result:
                key = f"{row.tablename}.{row.indexname}"
                usage[key] = float(row.idx_scan or 0)
                
        except Exception:
            # SQLiteの場合は固定値
            usage = {
                'users.ix_users_email': 95.0,
                'users.ix_users_username': 87.0,
                'posts.ix_posts_author_id': 92.0
            }
        
        return usage
    
    def get_cache_hit_ratio(self, session: Session) -> float:
        """キャッシュヒット率を取得"""
        try:
            # PostgreSQLの例
            result = session.exec(text("""
                SELECT 
                    sum(heap_blks_hit) / (sum(heap_blks_hit) + sum(heap_blks_read)) * 100
                FROM pg_statio_user_tables
            """)).one()
            return float(result or 0)
        except Exception:
            return 95.0  # デフォルト値
    
    def get_system_metrics(self) -> tuple[float, float, float]:
        """システムメトリクスを取得"""
        disk_usage = psutil.disk_usage('/').percent
        cpu_usage = psutil.cpu_percent(interval=1)
        memory_usage = psutil.virtual_memory().percent
        
        return disk_usage, cpu_usage, memory_usage
    
    def collect_metrics(self) -> DatabaseMetrics:
        """全メトリクスを収集"""
        with Session(self.engine) as session:
            # データベースメトリクス
            active_connections = self.get_connection_count(session)
            slow_queries = self.get_slow_queries(session)
            table_sizes = self.get_table_sizes(session)
            index_usage = self.get_index_usage(session)
            cache_hit_ratio = self.get_cache_hit_ratio(session)
            
            # システムメトリクス
            disk_usage, cpu_usage, memory_usage = self.get_system_metrics()
            
            return DatabaseMetrics(
                timestamp=datetime.utcnow(),
                active_connections=active_connections,
                slow_queries=slow_queries,
                table_sizes=table_sizes,
                index_usage=index_usage,
                cache_hit_ratio=cache_hit_ratio,
                disk_usage=disk_usage,
                cpu_usage=cpu_usage,
                memory_usage=memory_usage
            )
    
    def check_health(self, metrics: DatabaseMetrics) -> List[str]:
        """健康状態をチェック"""
        warnings = []
        
        # 接続数チェック
        if metrics.active_connections > 100:
            warnings.append(f"高い接続数: {metrics.active_connections}")
        
        # 遅いクエリチェック
        if metrics.slow_queries > 10:
            warnings.append(f"遅いクエリが多い: {metrics.slow_queries}")
        
        # キャッシュヒット率チェック
        if metrics.cache_hit_ratio < 90:
            warnings.append(f"低いキャッシュヒット率: {metrics.cache_hit_ratio:.1f}%")
        
        # システムリソースチェック
        if metrics.disk_usage > 85:
            warnings.append(f"ディスク使用率が高い: {metrics.disk_usage:.1f}%")
        
        if metrics.cpu_usage > 80:
            warnings.append(f"CPU使用率が高い: {metrics.cpu_usage:.1f}%")
        
        if metrics.memory_usage > 90:
            warnings.append(f"メモリ使用率が高い: {metrics.memory_usage:.1f}%")
        
        return warnings
    
    def run_monitoring_loop(self, interval: int = 60):
        """監視ループを実行"""
        self.logger.info("データベース監視を開始します")
        
        while True:
            try:
                metrics = self.collect_metrics()
                warnings = self.check_health(metrics)
                
                # ログ出力
                self.logger.info(f"監視データ収集完了: {metrics.timestamp}")
                self.logger.info(f"接続数: {metrics.active_connections}")
                self.logger.info(f"キャッシュヒット率: {metrics.cache_hit_ratio:.1f}%")
                self.logger.info(f"CPU: {metrics.cpu_usage:.1f}%, メモリ: {metrics.memory_usage:.1f}%, ディスク: {metrics.disk_usage:.1f}%")
                
                # 警告がある場合
                if warnings:
                    self.logger.warning("健康状態の警告:")
                    for warning in warnings:
                        self.logger.warning(f"  - {warning}")
                
                # メトリクスを保存（実際の実装では時系列データベースに保存）
                self.save_metrics(metrics)
                
            except Exception as e:
                self.logger.error(f"監視エラー: {e}")
            
            time.sleep(interval)
