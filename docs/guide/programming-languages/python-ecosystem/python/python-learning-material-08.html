実習 8-1: ファイルバックアップツール

指定されたディレクトリのファイルを日付付きでバックアップするツールを作成しましょう。

コード例
import os
import shutil
from datetime import datetime

class FileBackupTool:
    def __init__(self, source_dir, backup_base_dir="backups"):
        self.source_dir = source_dir
        self.backup_base_dir = backup_base_dir
        
        # バックアップディレクトリを作成
        os.makedirs(backup_base_dir, exist_ok=True)
    
    def create_backup(self):
        """バックアップを作成"""
        if not os.path.exists(self.source_dir):
            print(f"エラー: ソースディレクトリ '{self.source_dir}' が存在しません")
            return False
        
        # タイムスタンプ付きバックアップディレクトリ名を生成
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_dir = os.path.join(self.backup_base_dir, f"backup_{timestamp}")
        
        try:
            # ディレクトリ全体をコピー
            shutil.copytree(self.source_dir, backup_dir)
            print(f"バックアップ完了: {backup_dir}")
            
            # バックアップ情報を記録
            self._create_backup_info(backup_dir)
            return True
            
        except Exception as e:
            print(f"バックアップエラー: {e}")
            return False
    
    def _create_backup_info(self, backup_dir):
        """バックアップ情報ファイルを作成"""
        info_file = os.path.join(backup_dir, "backup_info.txt")
        
        with open(info_file, "w", encoding="utf-8") as f:
            f.write(f"バックアップ作成日時: {datetime.now()}\n")
            f.write(f"ソースディレクトリ: {self.source_dir}\n")
            f.write(f"バックアップディレクトリ: {backup_dir}\n\n")
            
            # ファイル一覧を記録
            f.write("=== ファイル一覧 ===\n")
            for root, dirs, files in os.walk(backup_dir):
                level = root.replace(backup_dir, '').count(os.sep)
                indent = ' ' * 2 * level
                f.write(f"{indent}{os.path.basename(root)}/\n")
                
                subindent = ' ' * 2 * (level + 1)
                for file in files:
                    if file != "backup_info.txt":  # 自分自身は除外
                        file_path = os.path.join(root, file)
                        file_size = os.path.getsize(file_path)
                        f.write(f"{subindent}{file} ({file_size} bytes)\n")
    
    def list_backups(self):
        """既存のバックアップ一覧を表示"""
        if not os.path.exists(self.backup_base_dir):
            print("バックアップディレクトリが存在しません")
            return
        
        backups = [d for d in os.listdir(self.backup_base_dir) 
                  if d.startswith("backup_") and os.path.isdir(
                      os.path.join(self.backup_base_dir, d))]
        
        if not backups:
            print("バックアップが見つかりません")
            return
        
        print("=== バックアップ一覧 ===")
        backups.sort(reverse=True)  # 新しい順
        
        for backup in backups:
            backup_path = os.path.join(self.backup_base_dir, backup)
            
            # バックアップサイズを計算
            total_size = 0
            for dirpath, dirnames, filenames in os.walk(backup_path):
                for filename in filenames:
                    filepath = os.path.join(dirpath, filename)
                    total_size += os.path.getsize(filepath)
            
            size_mb = total_size / (1024 * 1024)
            print(f"{backup}: {size_mb:.2f} MB")

# 使用例
# サンプルディレクトリとファイルを作成
sample_dir = "sample_data"
os.makedirs(sample_dir, exist_ok=True)

with open(os.path.join(sample_dir, "file1.txt"), "w") as f:
    f.write("サンプルファイル1の内容")

with open(os.path.join(sample_dir, "file2.txt"), "w") as f:
    f.write("サンプルファイル2の内容")

# バックアップツールを使用
backup_tool = FileBackupTool(sample_dir)
backup_tool.create_backup()
backup_tool.list_backups()

実習 8-2: モンテカルロ法による円周率の近似計算

乱数を使って円周率を近似計算するプログラムを作成しましょう。

コード例
import random
import math
import time

def monte_carlo_pi(num_points):
    """モンテカルロ法で円周率を近似計算"""
    inside_circle = 0
    
    for _ in range(num_points):
        # -1から1の範囲で乱数を生成
        x = random.uniform(-1, 1)
        y = random.uniform(-1, 1)
        
        # 原点からの距離が1以下なら円の内部
        if x*x + y*y <= 1:
            inside_circle += 1
    
    # π = 4 * (円内の点数 / 全体の点数)
    pi_approximation = 4 * inside_circle / num_points
    return pi_approximation, inside_circle

def run_pi_calculation():
    """円周率計算のデモンストレーション"""
    print("=== モンテカルロ法による円周率近似計算 ===")
    print(f"実際の円周率: {math.pi:.10f}")
    print()
    
    test_cases = [1000, 10000, 100000, 1000000]
    
    for num_points in test_cases:
        start_time = time.time()
        pi_approx, inside_count = monte_carlo_pi(num_points)
        end_time = time.time()
        
        error = abs(pi_approx - math.pi)
        error_percent = (error / math.pi) * 100
        
        print(f"試行回数: {num_points:,}")
        print(f"近似値: {pi_approx:.10f}")
        print(f"誤差: {error:.10f} ({error_percent:.4f}%)")
        print(f"計算時間: {end_time - start_time:.4f}秒")
        print(f"円内の点: {inside_count:,} / {num_points:,}")
        print("-" * 50)

# 実行
run_pi_calculation()

# 統計的な分析
def analyze_pi_distribution(trials=100, points_per_trial=10000):
    """複数回の計算結果の統計分析"""
    results = []
    
    print(f"\n=== 統計分析 ({trials}回の試行) ===")
    
    for _ in range(trials):
        pi_approx, _ = monte_carlo_pi(points_per_trial)
        results.append(pi_approx)
    
    # 統計情報の計算
    mean_pi = sum(results) / len(results)
    min_pi = min(results)
    max_pi = max(results)
    
    # 標準偏差の計算
    variance = sum((x - mean_pi) ** 2 for x in results) / len(results)
    std_dev = math.sqrt(variance)
    
    print(f"平均値: {mean_pi:.10f}")
    print(f"最小値: {min_pi:.10f}")
    print(f"最大値: {max_pi:.10f}")
    print(f"標準偏差: {std_dev:.10f}")
    print(f"実際のπとの差: {abs(mean_pi - math.pi):.10f}")

8.4 reモジュール - 正規表現

正規表現は、文字列のパターンマッチングを行うための強力なツールです。

正規表現の基本パターン

import re

# 基本的なパターンマッチング
text = "私の電話番号は090-1234-5678です。メールは user@example.com です。"

# パターンの検索
phone_pattern = r'\d{3}-\d{4}-\d{4}'
email_pattern = r'[\w\.-]+@[\w\.-]+\.\w+'

phone_match = re.search(phone_pattern, text)
email_match = re.search(email_pattern, text)

if phone_match:
    print(f"電話番号を発見: {phone_match.group()}")

if email_match:
    print(f"メールアドレスを発見: {email_match.group()}")

# 全てのマッチを検索
numbers_text = "価格は1,000円、税込み1,100円、送料500円です。"
price_pattern = r'\d{1,3}(?:,\d{3})*'
prices = re.findall(price_pattern, numbers_text)
print(f"価格一覧: {prices}")

# パターンの置換
masked_text = re.sub(phone_pattern, "XXX-XXXX-XXXX", text)
print(f"マスク済みテキスト: {masked_text}")

# 文字列の分割
csv_line = "名前,年齢,職業,住所"
fields = re.split(r',', csv_line)
print(f"CSV フィールド: {fields}")

# より複雑な分割（複数の区切り文字）
mixed_text = "apple;banana,orange:grape"
fruits = re.split(r'[;,:]+', mixed_text)
print(f"果物: {fruits}")

グループ化とキャプチャ

import re

# グループ化によるデータ抽出
log_text = """
2024-08-02 10:30:15 [INFO] ユーザーログイン: user123
2024-08-02 10:31:20 [ERROR] ファイルが見つかりません: data.txt
2024-08-02 10:32:05 [WARNING] メモリ使用量が高くなっています
"""

# ログパターンの定義（グループを使用）
log_pattern = r'(\d{4}-\d{2}-\d{2}) (\d{2}:\d{2}:\d{2}) \[(\w+)\] (.*)'

logs = []
for line in log_text.strip().split('\n'):
    match = re.match(log_pattern, line)
    if match:
        date, time, level, message = match.groups()
        logs.append({
            'date': date,
            'time': time,
            'level': level,
            'message': message
        })

print("=== ログ解析結果 ===")
for log in logs:
    print(f"{log['date']} {log['time']} [{log['level']}] {log['message']}")

# 名前付きグループ
url_pattern = r'https?://(?P[\w\.-]+)(?P/[\w\./]*)?(?P\?[\w=&]*)?'
url = "https://www.example.com/search/results?q=python&type=tutorial"

match = re.match(url_pattern, url)
if match:
    print(f"\nURL解析:")
    print(f"ドメイン: {match.group('domain')}")

# バリデーション関数
def validate_email(email):
    """メールアドレスの妥当性をチェック"""
    pattern = r'^[\w\.-]+@[\w\.-]+\.\w{2,}$'
    return bool(re.match(pattern, email))

def validate_phone(phone):
    """日本の電話番号の妥当性をチェック"""
    # 固定電話または携帯電話
    pattern = r'^(0\d{1,4}-\d{1,4}-\d{4}|0\d{9,10})$'
    return bool(re.match(pattern, phone))

# バリデーションテスト
test_emails = ["user@example.com", "invalid-email", "test@test.co.jp"]
test_phones = ["090-1234-5678", "03-1234-5678", "invalid-phone"]

print("\n=== バリデーション結果 ===")
for email in test_emails:
    result = "有効" if validate_email(email) else "無効"
    print(f"メール {email}: {result}")

for phone in test_phones:
    result = "有効" if validate_phone(phone) else "無効"
    print(f"電話 {phone}: {result}")
