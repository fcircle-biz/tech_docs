6.1 継承（Inheritance）

継承は、既存のクラスを土台にして新しいクラスを作成する仕組みです。共通の特徴を持つクラス群を効率的に設計でき、コードの再利用性と保守性が向上します。

classDiagram class Animal { +Name: String +Age: Integer +Eat() +Sleep() } class Dog { +Breed: String +Bark() } class Cat { +Color: String +Meow() } Animal <|-- Dog Animal <|-- Cat 
' 基底クラス（親クラス）
Public Class Animal
    Protected _name As String
    Protected _age As Integer
    
    Public Sub New(name As String, age As Integer)
        _name = name
        _age = age
    End Sub
    
    Public Overridable Sub MakeSound()
        Console.WriteLine($"{_name}が音を出しています")
    End Sub
    
    Public Sub Sleep()
        Console.WriteLine($"{_name}が眠っています")
    End Sub
End Class

' 派生クラス（子クラス）
Public Class Dog
    Inherits Animal
    
    Private _breed As String
    
    Public Sub New(name As String, age As Integer, breed As String)
        MyBase.New(name, age)  ' 基底クラスのコンストラクタを呼び出し
        _breed = breed
    End Sub
    
    Public Overrides Sub MakeSound()
        Console.WriteLine($"{_name}がワンワンと鳴いています")
    End Sub
    
    Public Sub Fetch()
        Console.WriteLine($"{_name}がボールを取ってきました")
    End Sub
End Class

6.2 ポリモーフィズム（Polymorphism）

ポリモーフィズムは、同じインターフェースを通じて異なる型のオブジェクトを操作できる仕組みです。基底クラスの参照で様々な派生クラスのオブジェクトを扱うことができます。 

Module Program
    Sub Main(args As String())
        ' ポリモーフィズムの例
        Dim animals() As Animal = {
            New Dog("ポチ", 3, "柴犬"),
            New Cat("タマ", 2, "白"),
            New Dog("ハチ", 5, "秋田犬")
        }
        
        Console.WriteLine("=== 動物園の動物たち ===\n")
        
        ' 同じメソッドでも、実際のオブジェクトによって異なる動作
        For Each animal As Animal In animals
            animal.MakeSound()  ' 各動物に応じた鳴き声
            animal.Sleep()      ' 共通の動作
            Console.WriteLine()
        Next
        
        Console.ReadKey()
    End Sub
End Module
実習 6-1: 図形クラス階層の作成

継承とポリモーフィズムを使って、様々な図形を管理するシステムを作成しましょう。

サンプルコード
' 基底クラス
Public MustInherit Class Shape
    Protected _name As String
    
    Public Sub New(name As String)
        _name = name
    End Sub
    
    Public ReadOnly Property Name As String
        Get
            Return _name
        End Get
    End Property
    
    ' 抽象メソッド - 派生クラスで必ず実装が必要
    Public MustOverride Function CalculateArea() As Double
    Public MustOverride Function CalculatePerimeter() As Double
    
    Public Sub ShowInfo()
        Console.WriteLine($"図形: {_name}")
        Console.WriteLine($"面積: {CalculateArea():F2}")
        Console.WriteLine($"周囲: {CalculatePerimeter():F2}")
        Console.WriteLine()
    End Sub
End Class

' 円クラス
Public Class Circle
    Inherits Shape
    
    Private _radius As Double
    
    Public Sub New(radius As Double)
        MyBase.New("円")
        _radius = radius
    End Sub
    
    Public Overrides Function CalculateArea() As Double
        Return Math.PI * _radius * _radius
    End Function
    
    Public Overrides Function CalculatePerimeter() As Double
        Return 2 * Math.PI * _radius
    End Function
End Class

' 長方形クラス
Public Class Rectangle
    Inherits Shape
    
    Private _width As Double
    Private _height As Double
    
    Public Sub New(width As Double, height As Double)
        MyBase.New("長方形")
        _width = width
        _height = height
    End Sub
    
    Public Overrides Function CalculateArea() As Double
        Return _width * _height
    End Function
    
    Public Overrides Function CalculatePerimeter() As Double
        Return 2 * (_width + _height)
    End Function
End Class

6.3 インターフェース（Interface）

インターフェースは、クラスが実装すべきメソッドやプロパティの契約を定義します。複数のクラスが同じインターフェースを実装することで、統一された操作が可能になります。 

' インターフェースの定義
Public Interface IVehicle
    Sub Start()
    Sub Stop()
    Property MaxSpeed As Integer
    ReadOnly Property FuelType As String
End Interface

' インターフェースの実装
Public Class Car
    Implements IVehicle
    
    Private _maxSpeed As Integer
    Private _isRunning As Boolean
    
    Public Sub New(maxSpeed As Integer)
        _maxSpeed = maxSpeed
        _isRunning = False
    End Sub
    
    Public Sub Start() Implements IVehicle.Start
        _isRunning = True
        Console.WriteLine("車のエンジンが始動しました")
    End Sub
    
    Public Sub Stop() Implements IVehicle.Stop
        _isRunning = False
        Console.WriteLine("車のエンジンが停止しました")
    End Sub
    
    Public Property MaxSpeed As Integer Implements IVehicle.MaxSpeed
        Get
            Return _maxSpeed
        End Get
        Set(value As Integer)
            _maxSpeed = value
        End Set
    End Property
    
    Public ReadOnly Property FuelType As String Implements IVehicle.FuelType
        Get
            Return "ガソリン"
        End Get
    End Property
End Class
