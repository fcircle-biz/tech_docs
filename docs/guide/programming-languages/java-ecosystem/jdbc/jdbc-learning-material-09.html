9.1 システム要件とデータベース設計

システム要件

今回構築する商品管理システムは、以下の機能を提供します：

  • 商品の新規登録（商品名、価格、在庫数、カテゴリ）

9.2 エンティティクラスの作成

データベースのproductsテーブルに対応するProductクラスを作成します。このクラスは商品情報を表現するモデルクラスです。

Product.java 
package com.example.model;

import java.math.BigDecimal;
import java.time.LocalDateTime;

/**
 * 商品情報を表現するエンティティクラス
 */
public class Product {
    private int productId;           // 商品ID
    private String productName;      // 商品名
    private BigDecimal price;        // 価格
    private int stockQuantity;       // 在庫数
    private String category;         // カテゴリ
    private LocalDateTime createdAt; // 登録日時
    private LocalDateTime updatedAt; // 更新日時

    // デフォルトコンストラクタ
    public Product() {}

    // 全フィールドを初期化するコンストラクタ
    public Product(int productId, String productName, BigDecimal price,
                   int stockQuantity, String category) {
        this.productId = productId;
        this.productName = productName;
        this.price = price;
        this.stockQuantity = stockQuantity;
        this.category = category;
    }

    // Getter/Setterメソッド
    public int getProductId() { return productId; }
    public void setProductId(int productId) { this.productId = productId; }

    public String getProductName() { return productName; }
    public void setProductName(String productName) { this.productName = productName; }

    public BigDecimal getPrice() { return price; }
    public void setPrice(BigDecimal price) { this.price = price; }

    public int getStockQuantity() { return stockQuantity; }
    public void setStockQuantity(int stockQuantity) { this.stockQuantity = stockQuantity; }

    public String getCategory() { return category; }
    public void setCategory(String category) { this.category = category; }

    public LocalDateTime getCreatedAt() { return createdAt; }
    public void setCreatedAt(LocalDateTime createdAt) { this.createdAt = createdAt; }

    public LocalDateTime getUpdatedAt() { return updatedAt; }
    public void setUpdatedAt(LocalDateTime updatedAt) { this.updatedAt = updatedAt; }

    @Override
    public String toString() {
        return String.format("Product[id=%d, name=%s, price=%s, stock=%d, category=%s]",
                productId, productName, price, stockQuantity, category);
    }
}

ProductDAO.java 
package com.example.dao;

import com.example.model.Product;
import java.math.BigDecimal;
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

/**
 * 商品テーブルへのデータアクセスを担当するDAOクラス
 */
public class ProductDAO {
    private final String url = "jdbc:postgresql://localhost:5432/productdb";
    private final String user = "postgres";
    private final String password = "password";

    /**
     * データベース接続を取得
     */
    private Connection getConnection() throws SQLException {
        return DriverManager.getConnection(url, user, password);
    }

    /**
     * 商品を新規登録
     */
    public boolean insertProduct(Product product) {
        String sql = "INSERT INTO products (product_name, price, stock_quantity, category) " +
                     "VALUES (?, ?, ?, ?)";

        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {

            pstmt.setString(1, product.getProductName());
            pstmt.setBigDecimal(2, product.getPrice());
            pstmt.setInt(3, product.getStockQuantity());
            pstmt.setString(4, product.getCategory());

            int rows = pstmt.executeUpdate();
            return rows > 0;

        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }

    /**
     * 全商品を取得
     */
    public List findAll() {
        List products = new ArrayList<>();
        String sql = "SELECT * FROM products ORDER BY product_id";

        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {

            while (rs.next()) {
                products.add(mapResultSetToProduct(rs));
            }

        } catch (SQLException e) {
            e.printStackTrace();
        }

        return products;
    }

    /**
     * IDで商品を検索
     */
    public Product findById(int productId) {
        String sql = "SELECT * FROM products WHERE product_id = ?";

        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {

            pstmt.setInt(1, productId);
            ResultSet rs = pstmt.executeQuery();

            if (rs.next()) {
                return mapResultSetToProduct(rs);
            }

        } catch (SQLException e) {
            e.printStackTrace();
        }

        return null;
    }

    /**
     * カテゴリで商品を検索
     */
    public List findByCategory(String category) {
        List products = new ArrayList<>();
        String sql = "SELECT * FROM products WHERE category = ? ORDER BY product_id";

        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {

            pstmt.setString(1, category);
            ResultSet rs = pstmt.executeQuery();

            while (rs.next()) {
                products.add(mapResultSetToProduct(rs));
            }

        } catch (SQLException e) {
            e.printStackTrace();
        }

        return products;
    }

    /**
     * 商品情報を更新
     */
    public boolean updateProduct(Product product) {
        String sql = "UPDATE products SET product_name = ?, price = ?, " +
                     "stock_quantity = ?, category = ?, updated_at = CURRENT_TIMESTAMP " +
                     "WHERE product_id = ?";

        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {

            pstmt.setString(1, product.getProductName());
            pstmt.setBigDecimal(2, product.getPrice());
            pstmt.setInt(3, product.getStockQuantity());
            pstmt.setString(4, product.getCategory());
            pstmt.setInt(5, product.getProductId());

            int rows = pstmt.executeUpdate();
            return rows > 0;

        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }

    /**
     * 在庫数を更新（トランザクション制御付き）
     */
    public boolean updateStock(int productId, int quantity) {
        Connection conn = null;

        try {
            conn = getConnection();
            conn.setAutoCommit(false); // トランザクション開始

            // 現在の在庫数を確認
            String selectSql = "SELECT stock_quantity FROM products WHERE product_id = ?";
            int currentStock = 0;

            try (PreparedStatement pstmt = conn.prepareStatement(selectSql)) {
                pstmt.setInt(1, productId);
                ResultSet rs = pstmt.executeQuery();

                if (rs.next()) {
                    currentStock = rs.getInt("stock_quantity");
                } else {
                    throw new SQLException("商品が見つかりません");
                }
            }

            // 在庫数を更新
            int newStock = currentStock + quantity;
            if (newStock < 0) {
                throw new SQLException("在庫数が不足しています");
            }

            String updateSql = "UPDATE products SET stock_quantity = ?, " +
                              "updated_at = CURRENT_TIMESTAMP WHERE product_id = ?";

            try (PreparedStatement pstmt = conn.prepareStatement(updateSql)) {
                pstmt.setInt(1, newStock);
                pstmt.setInt(2, productId);
                pstmt.executeUpdate();
            }

            conn.commit(); // コミット
            return true;

        } catch (SQLException e) {
            if (conn != null) {
                try {
                    conn.rollback(); // ロールバック
                } catch (SQLException ex) {
                    ex.printStackTrace();
                }
            }
            e.printStackTrace();
            return false;
        } finally {
            if (conn != null) {
                try {
                    conn.setAutoCommit(true);
                    conn.close();
                } catch (SQLException e) {
                    e.printStackTrace();
                }
            }
        }
    }

    /**
     * 商品を削除
     */
    public boolean deleteProduct(int productId) {
        String sql = "DELETE FROM products WHERE product_id = ?";

        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {

            pstmt.setInt(1, productId);
            int rows = pstmt.executeUpdate();
            return rows > 0;

        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }

    /**
     * ResultSetからProductオブジェクトへマッピング
     */
    private Product mapResultSetToProduct(ResultSet rs) throws SQLException {
        Product product = new Product();
        product.setProductId(rs.getInt("product_id"));
        product.setProductName(rs.getString("product_name"));
        product.setPrice(rs.getBigDecimal("price"));
        product.setStockQuantity(rs.getInt("stock_quantity"));
        product.setCategory(rs.getString("category"));
        product.setCreatedAt(rs.getTimestamp("created_at").toLocalDateTime());
        product.setUpdatedAt(rs.getTimestamp("updated_at").toLocalDateTime());
        return product;
    }
}

9.4 メインプログラムの作成

DAOクラスを使用して商品管理システムを操作するメインプログラムを作成します。

ProductManagementApp.java 
package com.example;

import com.example.dao.ProductDAO;
import com.example.model.Product;
import java.math.BigDecimal;
import java.util.List;
import java.util.Scanner;

/**
 * 商品管理システムのメインプログラム
 */
public class ProductManagementApp {
    private static ProductDAO productDAO = new ProductDAO();
    private static Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        System.out.println("=== 商品管理システム ===");

        while (true) {
            displayMenu();
            int choice = getIntInput("選択してください: ");

            switch (choice) {
                case 1:
                    registerProduct();
                    break;
                case 2:
                    listAllProducts();
                    break;
                case 3:
                    searchProductById();
                    break;
                case 4:
                    searchProductsByCategory();
                    break;
                case 5:
                    updateProductInfo();
                    break;
                case 6:
                    updateProductStock();
                    break;
                case 7:
                    deleteProduct();
                    break;
                case 0:
                    System.out.println("システムを終了します。");
                    return;
                default:
                    System.out.println("無効な選択です。");
            }

            System.out.println();
        }
    }

    private static void displayMenu() {
        System.out.println("\n--- メニュー ---");
        System.out.println("1. 商品登録");
        System.out.println("2. 全商品表示");
        System.out.println("3. 商品検索（ID）");
        System.out.println("4. 商品検索（カテゴリ）");
        System.out.println("5. 商品情報更新");
        System.out.println("6. 在庫数変更");
        System.out.println("7. 商品削除");
        System.out.println("0. 終了");
    }

    private static void registerProduct() {
        System.out.println("\n--- 商品登録 ---");

        String name = getStringInput("商品名: ");
        BigDecimal price = getBigDecimalInput("価格: ");
        int stock = getIntInput("在庫数: ");
        String category = getStringInput("カテゴリ: ");

        Product product = new Product();
        product.setProductName(name);
        product.setPrice(price);
        product.setStockQuantity(stock);
        product.setCategory(category);

        if (productDAO.insertProduct(product)) {
            System.out.println("商品を登録しました。");
        } else {
            System.out.println("商品の登録に失敗しました。");
        }
    }

    private static void listAllProducts() {
        System.out.println("\n--- 全商品一覧 ---");
        List products = productDAO.findAll();

        if (products.isEmpty()) {
            System.out.println("商品が登録されていません。");
        } else {
            for (Product p : products) {
                System.out.println(p);
            }
        }
    }

    private static void searchProductById() {
        System.out.println("\n--- 商品検索（ID） ---");
        int id = getIntInput("商品ID: ");

        Product product = productDAO.findById(id);
        if (product != null) {
            System.out.println(product);
        } else {
            System.out.println("商品が見つかりませんでした。");
        }
    }

    private static void searchProductsByCategory() {
        System.out.println("\n--- 商品検索（カテゴリ） ---");
        String category = getStringInput("カテゴリ: ");

        List products = productDAO.findByCategory(category);
        if (products.isEmpty()) {
            System.out.println("該当する商品が見つかりませんでした。");
        } else {
            for (Product p : products) {
                System.out.println(p);
            }
        }
    }

    private static void updateProductInfo() {
        System.out.println("\n--- 商品情報更新 ---");
        int id = getIntInput("商品ID: ");

        Product product = productDAO.findById(id);
        if (product == null) {
            System.out.println("商品が見つかりませんでした。");
            return;
        }

        System.out.println("現在の情報: " + product);

        String name = getStringInput("新しい商品名 (変更しない場合はEnter): ");
        if (!name.isEmpty()) {
            product.setProductName(name);
        }

        String priceStr = getStringInput("新しい価格 (変更しない場合はEnter): ");
        if (!priceStr.isEmpty()) {
            product.setPrice(new BigDecimal(priceStr));
        }

        String categoryStr = getStringInput("新しいカテゴリ (変更しない場合はEnter): ");
        if (!categoryStr.isEmpty()) {
            product.setCategory(categoryStr);
        }

        if (productDAO.updateProduct(product)) {
            System.out.println("商品情報を更新しました。");
        } else {
            System.out.println("商品情報の更新に失敗しました。");
        }
    }

    private static void updateProductStock() {
        System.out.println("\n--- 在庫数変更 ---");
        int id = getIntInput("商品ID: ");
        int quantity = getIntInput("増減数（減らす場合は負の数）: ");

        if (productDAO.updateStock(id, quantity)) {
            System.out.println("在庫数を更新しました。");
        } else {
            System.out.println("在庫数の更新に失敗しました。");
        }
    }

    private static void deleteProduct() {
        System.out.println("\n--- 商品削除 ---");
        int id = getIntInput("商品ID: ");

        if (productDAO.deleteProduct(id)) {
            System.out.println("商品を削除しました。");
        } else {
            System.out.println("商品の削除に失敗しました。");
        }
    }

    // ユーティリティメソッド
    private static String getStringInput(String prompt) {
        System.out.print(prompt);
        return scanner.nextLine();
    }

    private static int getIntInput(String prompt) {
        while (true) {
            try {
                System.out.print(prompt);
                return Integer.parseInt(scanner.nextLine());
            } catch (NumberFormatException e) {
                System.out.println("数値を入力してください。");
            }
        }
    }

    private static BigDecimal getBigDecimalInput(String prompt) {
        while (true) {
            try {
                System.out.print(prompt);
                return new BigDecimal(scanner.nextLine());
            } catch (NumberFormatException e) {
                System.out.println("正しい数値を入力してください。");
            }
        }
    }
}

実践的なベストプラクティス

  • 設定の外部化: データベース接続情報をプロパティファイルに分離する
  • ログ記録: エラー情報をログファイルに記録し、トラブルシューティングを容易にする
  • 入力検証: ユーザー入力を検証し、不正なデータの登録を防ぐ
  • コネクションプール: 本番環境ではコネクションプールを使用してパフォーマンスを向上させる
  • 単体テスト: JUnitを使用してDAOクラスのメソッドをテストする

理解度確認クイズ

  1. Q1. DAOパターンを使用する主な利点は何ですか？

    回答を表示

    DAOパターンの主な利点は、データアクセスロジックをビジネスロジックから分離できることです。これにより、コードの保守性が向上し、データベースの変更があってもDAOクラスだけを修正すればよくなります。また、テストが容易になり、複数の開発者が並行して作業しやすくなります。

  2. Q2. updateStockメソッドでトランザクション制御を行っている理由を説明してください。

    回答を表示

    updateStockメソッドでは、在庫数の確認と更新という2つの操作を1つの処理単位として扱う必要があります。トランザクション制御により、在庫確認後に別の処理が在庫を変更してしまう問題を防ぎ、データの整合性を保証します。また、在庫不足などのエラー発生時に確実にロールバックできます。

  3. Q3. mapResultSetToProductメソッドのような共通メソッドを作成する利点は何ですか？

    回答を表示

    ResultSetからエンティティへのマッピング処理を共通化することで、コードの重複を避け、保守性が向上します。カラム名の変更やデータ型の変更があった場合、1箇所の修正で全ての検索メソッドに反映されます。また、バグが発生した際も修正箇所が1つで済むため、品質向上にもつながります。

  4. Q4. 本番環境で商品管理システムを運用する際に追加すべきセキュリティ対策を3つ挙げてください。

    回答を表示

    1) 認証・認可: ユーザー認証機能を追加し、適切な権限を持つユーザーのみがシステムにアクセスできるようにする。
    2) 入力検証の強化: SQLインジェクションやXSSなどの攻撃を防ぐため、入力値の厳密なバリデーションを実装する。
    3) 接続情報の暗号化: データベース接続情報を暗号化し、環境変数や秘密鍵管理システムを使用して安全に管理する。
