実習 10-1: 個人家計簿アプリケーション

VB.NETの学習内容を全て活用して、実用的な個人家計簿アプリケーションを作成します。

アプリケーション仕様
  • 収入・支出の記録機能
  • カテゴリ別の分類
  • 月別・期間別の統計表示
  • データのCSVファイル保存・読込
  • 検索・フィルタリング機能
メインプログラム構造
Imports System.IO
Imports System.Globalization

Module Program
    Private budgetManager As New BudgetManager()
    
    Sub Main(args As String())
        Console.WriteLine("=== 個人家計簿アプリケーション ===")
        Console.WriteLine("VB.NET学習の集大成プロジェクト")
        Console.WriteLine()
        
        Try
            ' データファイルの読み込み
            budgetManager.LoadFromFile("budget_data.csv")
            Console.WriteLine("既存データを読み込みました。")
        Catch ex As FileNotFoundException
            Console.WriteLine("新しい家計簿を開始します。")
        Catch ex As Exception
            Console.WriteLine("データ読み込みエラー: " & ex.Message)
        End Try
        
        ' メインループ
        Do
            ShowMainMenu()
            Dim choice As String = Console.ReadLine()
            
            Select Case choice
                Case "1"
                    AddTransaction()
                Case "2"
                    ViewTransactions()
                Case "3"
                    ShowStatistics()
                Case "4"
                    SearchTransactions()
                Case "5"
                    SaveData()
                Case "0"
                    Exit Do
                Case Else
                    Console.WriteLine("無効な選択です。")
            End Select
            
            Console.WriteLine()
        Loop
        
        ' 終了時の自動保存
        Try
            budgetManager.SaveToFile("budget_data.csv")
            Console.WriteLine("データを保存しました。ありがとうございました！")
        Catch ex As Exception
            Console.WriteLine("保存エラー: " & ex.Message)
        End Try
    End Sub
    
    Sub ShowMainMenu()
        Console.WriteLine("=== メニュー ===")
        Console.WriteLine("1. 収支を記録する")
        Console.WriteLine("2. 取引履歴を表示する")
        Console.WriteLine("3. 統計情報を見る")
        Console.WriteLine("4. 取引を検索する")
        Console.WriteLine("5. データを保存する")
        Console.WriteLine("0. 終了する")
        Console.Write("選択してください (0-5): ")
    End Sub
    
    Sub AddTransaction()
        Console.WriteLine("=== 収支の記録 ===")
        
        Try
            ' 日付入力
            Console.Write("日付 (yyyy/MM/dd, 空白で今日): ")
            Dim dateInput As String = Console.ReadLine()
            Dim transactionDate As DateTime
            
            If String.IsNullOrWhiteSpace(dateInput) Then
                transactionDate = DateTime.Today
            Else
                transactionDate = DateTime.ParseExact(dateInput, "yyyy/MM/dd", CultureInfo.InvariantCulture)
            End If
            
            ' カテゴリ選択
            Console.WriteLine("カテゴリを選択:")
            Console.WriteLine("1.食費 2.交通費 3.娯楽 4.光熱費 5.その他支出 6.給与 7.その他収入")
            Console.Write("選択 (1-7): ")
            Dim categoryChoice As Integer = Integer.Parse(Console.ReadLine())
            
            Dim categories() As String = {"食費", "交通費", "娯楽", "光熱費", "その他支出", "給与", "その他収入"}
            Dim category As String = categories(categoryChoice - 1)
            
            ' 金額入力
            Console.Write("金額: ")
            Dim amount As Decimal = Decimal.Parse(Console.ReadLine())
            
            ' 説明入力
            Console.Write("説明 (省略可): ")
            Dim description As String = Console.ReadLine()
            If String.IsNullOrWhiteSpace(description) Then
                description = category
            End If
            
            ' 収支判定（カテゴリ6,7は収入）
            Dim isIncome As Boolean = (categoryChoice >= 6)
            
            ' トランザクション作成・追加
            Dim transaction As New Transaction(transactionDate, category, amount, description, isIncome)
            budgetManager.AddTransaction(transaction)
            
            Console.WriteLine("✓ 記録しました！")
            
        Catch ex As Exception
            Console.WriteLine("入力エラー: " & ex.Message)
        End Try
    End Sub
    
    Sub ViewTransactions()
        Console.WriteLine("=== 取引履歴 ===")
        Dim transactions = budgetManager.GetAllTransactions()
        
        If transactions.Count = 0 Then
            Console.WriteLine("まだ取引がありません。")
            Return
        End If
        
        Console.WriteLine("日付".PadRight(12) & "カテゴリ".PadRight(10) & "金額".PadRight(12) & "説明")
        Console.WriteLine("".PadRight(50, "-"c))
        
        For Each transaction In transactions.OrderByDescending(Function(t) t.Date)
            Dim amountStr As String = If(transaction.IsIncome, "+", "") & transaction.Amount.ToString("N0")
            Console.WriteLine($"{transaction.Date:yyyy/MM/dd}".PadRight(12) &
                            transaction.Category.PadRight(10) &
                            amountStr.PadLeft(10) &
                            $"   {transaction.Description}")
        Next
    End Sub
    
    Sub ShowStatistics()
        Console.WriteLine("=== 統計情報 ===")
        Dim stats = budgetManager.CalculateStatistics()
        
        Console.WriteLine($"総収入: {stats.TotalIncome:C}")
        Console.WriteLine($"総支出: {stats.TotalExpense:C}")
        Console.WriteLine($"収支: {stats.Balance:C}")
        Console.WriteLine()
        
        Console.WriteLine("カテゴリ別支出:")
        For Each kvp In stats.ExpenseByCategory.OrderByDescending(Function(x) x.Value)
            Console.WriteLine($"  {kvp.Key}: {kvp.Value:C}")
        Next
    End Sub
    
    Sub SearchTransactions()
        Console.WriteLine("=== 取引検索 ===")
        Console.Write("カテゴリ名を入力 (部分一致): ")
        Dim searchTerm As String = Console.ReadLine()
        
        Dim results = budgetManager.SearchByCategory(searchTerm)
        
        If results.Count = 0 Then
            Console.WriteLine("該当する取引が見つかりませんでした。")
            Return
        End If
        
        Console.WriteLine($"「{searchTerm}」の検索結果 ({results.Count}件):")
        For Each transaction In results.OrderByDescending(Function(t) t.Date)
            Console.WriteLine($"{transaction.Date:yyyy/MM/dd} {transaction.Category} {transaction.Amount:C} - {transaction.Description}")
        Next
    End Sub
    
    Sub SaveData()
        Try
            budgetManager.SaveToFile("budget_data.csv")
            Console.WriteLine("✓ データを保存しました。")
        Catch ex As Exception
            Console.WriteLine("保存エラー: " & ex.Message)
        End Try
    End Sub
End Module

10.2 データモデルとビジネスロジック

オブジェクト指向設計の原則に従い、データモデルとビジネスロジックを分離した設計を実装します。これにより、保守性と拡張性の高いアプリケーションを作成できます。 

' トランザクションデータモデル
Public Class Transaction
    Public Property Id As Guid
    Public Property Date As DateTime
    Public Property Category As String
    Public Property Amount As Decimal
    Public Property Description As String
    Public Property IsIncome As Boolean
    
    Public Sub New()
        Id = Guid.NewGuid()
        Date = DateTime.Today
    End Sub
    
    Public Sub New(date As DateTime, category As String, amount As Decimal, description As String, isIncome As Boolean)
        Me.New()
        Me.Date = date
        Me.Category = category
        Me.Amount = amount
        Me.Description = description
        Me.IsIncome = isIncome
    End Sub
    
    Public Overrides Function ToString() As String
        Dim type As String = If(IsIncome, "収入", "支出")
        Return $"{Date:yyyy/MM/dd} [{type}] {Category}: {Amount:C} - {Description}"
    End Function
End Class

' 統計情報クラス
Public Class BudgetStatistics
    Public Property TotalIncome As Decimal
    Public Property TotalExpense As Decimal
    Public Property Balance As Decimal
    Public Property ExpenseByCategory As Dictionary(Of String, Decimal)
    Public Property IncomeByCategory As Dictionary(Of String, Decimal)
    
    Public Sub New()
        ExpenseByCategory = New Dictionary(Of String, Decimal)
        IncomeByCategory = New Dictionary(Of String, Decimal)
    End Sub
End Class

' ビジネスロジッククラス
Public Class BudgetManager
    Private transactions As List(Of Transaction)
    
    Public Sub New()
        transactions = New List(Of Transaction)
    End Sub
    
    Public Sub AddTransaction(transaction As Transaction)
        If transaction Is Nothing Then
            Throw New ArgumentNullException("transaction")
        End If
        
        If transaction.Amount <= 0 Then
            Throw New ArgumentException("金額は正の数である必要があります")
        End If
        
        transactions.Add(transaction)
    End Sub
    
    Public Function GetAllTransactions() As List(Of Transaction)
        Return New List(Of Transaction)(transactions)
    End Function
    
    Public Function SearchByCategory(searchTerm As String) As List(Of Transaction)
        If String.IsNullOrWhiteSpace(searchTerm) Then
            Return New List(Of Transaction)
        End If
        
        Return transactions.Where(Function(t) t.Category.Contains(searchTerm)).ToList()
    End Function
    
    Public Function GetTransactionsByDateRange(startDate As DateTime, endDate As DateTime) As List(Of Transaction)
        Return transactions.Where(Function(t) t.Date >= startDate AndAlso t.Date <= endDate).ToList()
    End Function
    
    Public Function CalculateStatistics() As BudgetStatistics
        Dim stats As New BudgetStatistics()
        
        For Each transaction In transactions
            If transaction.IsIncome Then
                stats.TotalIncome += transaction.Amount
                
                If stats.IncomeByCategory.ContainsKey(transaction.Category) Then
                    stats.IncomeByCategory(transaction.Category) += transaction.Amount
                Else
                    stats.IncomeByCategory.Add(transaction.Category, transaction.Amount)
                End If
            Else
                stats.TotalExpense += transaction.Amount
                
                If stats.ExpenseByCategory.ContainsKey(transaction.Category) Then
                    stats.ExpenseByCategory(transaction.Category) += transaction.Amount
                Else
                    stats.ExpenseByCategory.Add(transaction.Category, transaction.Amount)
                End If
            End If
        Next
        
        stats.Balance = stats.TotalIncome - stats.TotalExpense
        Return stats
    End Function
    
    Public Sub SaveToFile(fileName As String)
        Using writer As New StreamWriter(fileName, False, Encoding.UTF8)
            ' ヘッダー行
            writer.WriteLine("ID,Date,Category,Amount,Description,IsIncome")
            
            ' データ行
            For Each transaction In transactions
                writer.WriteLine($"{transaction.Id},{transaction.Date:yyyy-MM-dd},{transaction.Category},{transaction.Amount},{transaction.Description},{transaction.IsIncome}")
            Next
        End Using
    End Sub
    
    Public Sub LoadFromFile(fileName As String)
        If Not File.Exists(fileName) Then
            Throw New FileNotFoundException("ファイルが見つかりません: " & fileName)
        End If
        
        transactions.Clear()
        
        Using reader As New StreamReader(fileName, Encoding.UTF8)
            ' ヘッダー行をスキップ
            reader.ReadLine()
            
            Dim line As String
            Do
                line = reader.ReadLine()
                If line IsNot Nothing Then
                    Dim parts() As String = line.Split(","c)
                    If parts.Length = 6 Then
                        Dim transaction As New Transaction() With {
                            .Id = Guid.Parse(parts(0)),
                            .Date = DateTime.Parse(parts(1)),
                            .Category = parts(2),
                            .Amount = Decimal.Parse(parts(3)),
                            .Description = parts(4),
                            .IsIncome = Boolean.Parse(parts(5))
                        }
                        transactions.Add(transaction)
                    End If
                End If
            Loop Until line Is Nothing
        End Using
    End Sub
End Class
